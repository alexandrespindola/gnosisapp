ގޭގައި ބިތްގަޑިއެއް ބަހައްޓާފައި ހުރުމަކީ، ހަމަ އެކަނި ގަޑި ބަލާލުމަށްވުރެ އިތުރަށް ވިސްނާ ފިކުރުކުރުމަށްވެސް އެހީތެރިވެދޭ ކަމަކަށް ވާތީ، އެކަން އެހާ ޝައުޤުވެރިވެއެވެ.

ޕެންޑިއުލަމް ނެތިއްޔާ ގަޑިއެއް ނުހިނގާނެއެވެ؛ ޕެންޑިއުލަމްގެ ހަރަކާތަކީ ވަރަށް ފުން މާނައެއް އެކުލެވިގެންވާ ކަމެކެވެ.

ކުރީ ޒަމާނުގަ އިވޮލިއުޝަންގެ ޑޮގްމާ އަކީ ވުޖޫދުވެސް އޮތް އެއްޗެއް ނޫނެވެ؛ އޭރު، ދުނިޔޭގެ ހެޔޮލަފާ މީހުން ދަނެއެވެ، ތާރީޚީ ކަންތައްތައް އަބަދުވެސް ކުރިއަށްދަނީ ޕެންޑިއުލަމްގެ ޤާނޫނުގެ ދަށުންކަން.

ހުރިހާ އެއްޗެއްވެސް ދައުރުވެ އަނބުރާ ރުޖޫޢަވެއެވެ، މައްޗަށް އަރާ ތިރިއަށް ފައިބައެވެ، ބޮޑުވެ ކުޑަވެއެވެ، މި ޤާނޫނުގެ ދަށުން ދެކޮޅަށް ދަތުރުކުރެއެވެ.

ހުރިހާ އެއްޗެއް އޮސިލޭޓްވާކަމާއި، ހުރިހާ އެއްޗެއް ވަގުތުގެ އޮއިވަރަށް ލެނބިފައިވާކަމާއި، ހުރިހާ އެއްޗެއް އިވޮލްވްވެ އިންވޮލްވްވާކަމީ އެއްވެސް އާశ్ޗަރިއުވުންވަނިވި ކަމެއް ނޫނެވެ.

ޕެންޑިއުލަމްގެ އެއްކޮޅުގައި އުފާވެރިކަން ވާއިރު އަނެއްކޮޅުގައި ވޭންވެރިކަންވެއެވެ؛ އަހަރެމެންގެ ހުރިހާ އިމޯޝަންތަކާއި، ޚިޔާލުތަކާއި، އެދުންތަކާއި، އުއްމީދުތައް ޕެންޑިއުލަމްގެ ޤާނޫނުގެ ދަށުން އޮސިލޭޓްވެއެވެ.

އުއްމީދާއި މާޔޫސްކަން، ނާމާންކަމާއި އުއްމީދު، ޝައުޤުވެރިކަމާއި ވޭން، ކާމިޔާބާއި ނާކާމިޔާބު، މޮޅާއި ގެއްލުން، ޔަޤީނުންވެސް މި ކަންތައްތައް މިވަނީ ޕެންޑިއުލާ ހަރަކާތުގެ ދެކޮޅުގައެވެ.

މިޞްރު އޭގެ އެންމެ ބޮޑު ބާރާއި ވެރިކަމާއެކު މުޤައްދަސް ކޯރެއްގެ އައްސޭރީގައި އުފެދުނެވެ، އެހެންނަމަވެސް ޕެންޑިއުލަމް އަނެއް ފަރާތަށް ދިޔައިމާ، އިދިކޮޅު ކޮޅުން މައްޗަށް އެރިތަނާ ފިރޢައުނުންގެ ބިން ވެއްޓިގެންދިޔައީއެވެ އަދި ނަބީބޭކަލުންގެ ލޮބުވެތި ޝަހަރު، ޖެރޫސަލަމް އުފެދުނެވެ.

ޕެންޑިއުލަމްގެ މަޤާމު ބަދަލުވުމުން އިސްރާއީލް ވެއްޓުނު އަދި އަނެއްކޮޅުން ރޫމީ އެމްޕަޔަރ އުފެދުނެވެ.

ޕެންޑިއުލާ ހަރަކާތުން އެމްޕަޔަރތައް މައްޗަށް އުފުލައި އަޑިއަށް ވައްޓާލައެވެ، ބާރުގަދަ ސިވިލައިޒޭޝަންތައް އުފައްދާފައި އަނެއްކާ އެތަންތަން ސުންނާފަތިކޮށްލައެވެ، އަދި މިފަދަ އެހެނިހެން ކަންތައްވެސް ކުރެއެވެ.

ޕެންޑިއުލަމްގެ ކަނާތްފަރާތުގައި ސޫޑޯ-އެސޮޓެރިކް އަދި ސޫޑޯ-އޮކަލްޓިސްޓް ސްކޫލްތަކާއި، ދީންތަކާއި، ސެކްޓްތައް ބެހެއްޓިދާނެއެވެ.

އަދި ޕެންޑިއުލަމްގެ ވާތްފަރާތުގައި ހުރިހާ މެޓީރިއަލިސްޓިކް ވައްތަރުގެ ސްކޫލްތަކާއި، މާކްސިސްޓުންނާއި، އެތީއިސްޓުންނާއި، ސިޕްޓިސިސްޓް ފަދަ މީހުން ބެހެއްޓިދާނެއެވެ. ޕެންޑިއުލާ ހަރަކާތުގެ އެންޓިތީސިސް، ބަދަލުވަމުންދާ، ހުއްޓުމެއްނެތި ތަފާތުވެ ބަދަލުވެއެވެ.

ދީނީ ގޮތުން ހައްދުފަހަނައަޅާފައިވާ މީހަކު، ކޮންމެ އާދަޔާ ޚިލާފު ކަމެއްގެ ސަބަބުން ނުވަތަ މާޔޫސްކަމެއްގެ ސަބަބުން ޕެންޑިއުލަމްގެ އަނެއްކޮޅަށް ގޮސް، އެތީއިސްޓަކަށް، މެޓީރިއަލިސްޓަކަށް، ސިޕްޓިކަކަށް ވެދާނެއެވެ.

މެޓީރިއަލިސްޓިކް، އެތީއިސްޓް، ހައްދުފަހަނައަޅާފައިވާ މީހަކު، ކޮންމެ އާދަޔާ ޚިލާފު ކަމެއްގެ ސަބަބުން، ފަހަރެއްގައި މެޓަފިޒިކަލް ޓްރާންސެންޑެންޓަލް ހާދިސާއެއްގެ ސަބަބުން ނުވަތަ ބަޔާންކުރަން ދަތި ބިރުވެރި ވަގުތެއްގެ ސަބަބުން ޕެންޑިއުލާ ހަރަކާތުގެ އިދިކޮޅު ކޮޅަށް ގެންގޮސް އެމީހާ އެކަހެރި، ބަދުނާމު ދީނީ މީހަކަށް ބަދަލުކޮށްފާނެއެވެ.

މިސާލު: އެސޮޓެރިސްޓަކު ބަހުސްކޮށްގެން ބަލިކުރި ފާދިރީއަކު މާޔޫސްވެގެން އީމާންކަން ދޫކޮށް މެޓީރިއަލިސްޓަކަށް ވެއްޖެއެވެ.

އަހަރެމެންނަށް އެތީއިސްޓް އަންހެނެއްގެ މައްސަލައެއް އެނގެއެވެ. މެޓަފިޒިކަލް ކަމެއްގެ ސަބަބުން އެކަނބުލޭގެ ވަނީ ޕްރެކްޓިކަލް އެސޮޓެރިޒަމްގެ މޮޅު ތަރުޖަމާނަކަށް ވެފައެވެ.

ހަޤީޤަތުގެ ނަމުގައި އަހަރެމެން ބުނަންޖެހެނީ ހަޤީޤީ އަދި މުޠުލަޤު މެޓީރިއަލިސްޓް އެތީއިސްޓަކީ ދޮގެއްކަމަށާއި، އެފަދަ އެއްޗެއް ނުވާކަމަށެވެ.

އަނިޔާވެރި މަރެއްގެ ކުރިމަތީގައި ނުވަތަ ބަޔާންކުރަން ދަތި ބިރުވެރިކަމެއްގެ ވަގުތުގައި އަބަދީ ޢަދުއްވުންނާއި، މެޓީރިއަލިސްޓުންނާއި އީމާންކަން ނެތް މީހުން ވަގުތުން ޕެންޑިއުލަމްގެ އަނެއްކޮޅަށްގޮސް ނިމިގެންދަނީ ނިހާޔަތް ނެތް އީމާންތެރިކަމާއި ބޮޑު އަޅުވެތިކަމާއެކު ދުޢާކޮށް، ރޮއި، އާހްއަޅުވަމުންނެވެ.

މެޓީރިއަލިޒަމް ޑައިއެލެކްޓިކްގެ މުއައްލިފު ކާލް މާކްސްވެސް ވީ ޔަހޫދީ ދީނުގެ ހައްދުފަހަނައަޅާފައިވާ މީހަކަށެވެ، އަދި އޭނާ މަރުވުމަށްފަހު، ރައްބީންގެ ވަރަށްބޮޑު އިޙްތިރާމަކާއެކު އޭނާގެ ޖަނާޒާގެ ކަންތައްތައް ކުރިއަށް ގެންދިޔައެވެ.

ކާލް މާކްސް އޭނާގެ މާއްދީ ޑިއަލެކްޓިކް އުފެއްދީ އެންމެ މަޤުޞަދެއްގައެވެ: "ދުނިޔޭގެ ހުރިހާ ދީންތަކެއް ސިޕްޓިސިޒަމް މެދުވެރިކޮށް ނެތިކޮށްލުމަށްޓަކައި ހަތިޔާރެއް އުފެއްދުން."

މިއީ ދީނީ ގޮތުން އީރަޝްވެރިވުން އިންތިހާއަށް ދިއުމުގެ ސަބަބުން ދިމާވާ ކަންތައްތަކުގެ މިސާލެކެވެ؛ އެހެން ދީންތަކެއް އޮތުމަކީ މާކްސް އެއްގޮތަކަށްވެސް ޤަބޫލުކުރާނެ ކަމެއްނޫނެވެ. އެހެންކަމުން އޭނާގެ ޑިއަލެކްޓިކް މެދުވެރިކޮށް އެ ދީންތައް ނެތިކޮށްލުމަށް އިސްކަންދިނެވެ.

ކާލް މާކްސް ވަނީ ޒަޔަންގެ ޕްރޮޓޮކޯލްތަކުގެ ތެރެއިން އެއް ޕްރޮޓޮކޯލް ފުރިހަމަކޮށްފައެވެ، އެ ޕްރޮޓޮކޯލްގައި ސީދާގޮތުން ބުނެފައިވަނީ: "އަހަރެމެން ދުނިޔެ މެޓީރިއަލިޒަމްއާއި ނަފްރަތު އުފައްދާ އެތީއިޒަމްއިން ފުރާލިޔަސް މައްސަލައެއްނެތް، އަހަރެމެންނަށް ކާމިޔާބު ލިބޭ ދުވަހު، އަހަރެމެން މޫސާގެފާނުގެ ދީން އެކަށީގެންވާ ގޮތަށް ކޯޑުކޮށް ޑިއަލެކްޓިކް ގޮތަކަށް ދަސްކޮށްދޭނަން، އަދި ދުނިޔޭގައި އެހެން އެއްވެސް ދީނަކަށް ފުރުޞަތު ނުދޭނަން."

ސޯވިއެޓް ޔޫނިއަންގައި ދީންތަކަށް ދަތިކުރަމުންދާކަމާއި، މާއްދީ ޑައިއެލެކްޓިކް ރައްޔިތުންނަށް ދަސްކޮށްދެމުންދާކަމީ، ވަރަށް ޝައުޤުވެރި ކަމެކެވެ، އެހެންނަމަވެސް ސިނަގޯގްތަކުގައި ތަލްމޫދާއި، ބައިބަލްއާއި، ދީން ދަސްކޮށްދެމުންދާކަމާއި، އެއްވެސް މައްސަލައެއްނެތި މިނިވަންކަމާއެކު މަސައްކަތްކުރަމުންދާކަމީވެސް ފާހަގަކޮށްލަންޖެހޭ ކަމެކެވެ.

ރަޝިޔާ ސަރުކާރުގެ ވެރިންނަކީ މޫސާގެފާނުގެ ޤާނޫނަށް ތަބާވާ ހައްދުފަހަނައަޅާފައިވާ ދީންވެރިންނެވެ، އެހެންނަމަވެސް އެމީހުން މާއްދީ ޑިއަލެކްޓިކްގެ ދޮގުބަސްތަކުން ރައްޔިތުންނަށް ވިހަދެއެވެ.

އަހަރެމެން ދުވަހަކުވެސް އިސްރާއީލްގެ ރައްޔިތުންނާ ދެކޮޅުވާނެ ގޮތަކަށް ވާހަކަ ނުދައްކާނަން؛ އަހަރެމެން މި ބަޔާންކުރަނީ ހަމައެކަނި ސިއްރު މަޤުޞަދުތަކެއް ހާސިލުކުރުމަށް މަސައްކަތްކުރާ، ދެގޮތް ދައްކާ ވަކި މީހުންގެ ބަޔަކާ ދެކޮޅަށެވެ، އެމީހުން މާއްދީ ޑިއަލެކްޓިކްގެ ސަބަބުން ރައްޔިތުންނަށް ވިހަދެނީ ސިއްރުން މޫސާގެފާނުގެ ދީނަށް ޢަމަލުކުރަމުންނެވެ.

މެޓީރިއަލިޒަމްއާއި ރޫހާނީ ޢިލްމާއި، އޭގެ ފަހަތުން އަންނަ ހުރިހާ ވައްތަރެއްގެ ތިއަރީތަކާއި، ޕްރިޖުޑިސްތަކާއި، ޕްރީ-ކޮންސެޕްޓްތައް ހިނގަނީ ޕެންޑިއުލަމްގެ ޤާނޫނުގެ ދަށުންނެވެ އަދި މިއީ ވަގުތާއި އާދަކާދަޔާ އެއްގޮތަށް ބަދަލުވަމުންދާ އެއްޗެކެވެ.

ރޫހާއި މާއްދާއަކީ ވަރަށް ބޮޑަށް ބަހުސްކުރެވޭ، އެއްވެސް މީހަކަށް ނުވިސްނޭ ދެކޮންސެޕްޓެވެ.

ރޫހާނީކަމާ ބެހޭގޮތުން ސިކުނޑިއަކަށް އެއްވެސް އެއްޗެއް ނޭނގޭ، މާއްދާއާ ބެހޭގޮތުންވެސް އެއްވެސް އެއްޗެއް ނޭނގޭ.

ކޮންސެޕްޓަކީ ހަމަ އެއީ އެންމެ ކޮންސެޕްޓެކެވެ. ހަޤީޤަތަކީ ކޮންސެޕްޓެއް ނޫނެވެ، އެހެންނަމަވެސް ހަޤީޤަތާމެދު ސިކުނޑިން އެތައް ކޮންސެޕްޓެއް އުފެއްދިދާނެއެވެ.

ރޫހަކީ ރޫހެވެ (ވުޖޫދެވެ)، އަދި ހަމައެކަނި އަމިއްލަ ނަފްސަށް އެނގޭނެ އެއްޗެކެވެ.

ލިޔެވިފައިވަނީ: "ވުޖޫދަކީ ވުޖޫދެވެ އަދި ވުޖޫދުގެ ސަބަބަކީ ހަމަ އެ ވުޖޫދެވެ."

މާއްދާގެ ކަޓްޓަރުންނާއި، މެޓީރިއަލިޒަމް ޑައިއެލެކްޓިކްގެ ސައިންސްވެރިންނަކީ ސައްތައިން ސައްތަ އިމްޕީރިކަލް ބައެކެވެ އަދި މޮޔަ ބައެކެވެ. އެމީހުން މާއްދާއާ ބެހޭގޮތުން ވަރަށް ބޮޑާކަމާއި މޮޔަކަމާއެކު ވާހަކަ ދައްކާއިރު ހަޤީޤަތުގައި އެމީހުންނަށް އެއްވެސް އެއްޗެއް ނޭނގެއެވެ.

މާއްދާއަކީ ކޮބައިހެއްޔ؟ މި މޮޔަ ސައިންސްވެރިންގެ ތެރެއިން އެކަން ދަންނަނީ ކާކުހެއްޔ؟ އެހާ މަޝްހޫރު މާއްދާއަކީވެސް ވަރަށް ބޮޑަށް ބަހުސްކުރެވޭ އަދި ތޫނު ކޮންސެޕްޓެކެވެ.

މާއްދާއަކީ ކޮބައިހެއްޔ؟، ފޮތިހެއްޔ؟، ދަގަނޑުހެއްޔ؟، މަސްހެއްޔ؟، ގޮދަންހެއްޔ؟، ގަލެއްހެއްޔ؟، ރަންހެއްޔ؟، ވިލާގަނޑެއްހެއްޔ؟ ނުވަތަ ކޮން އެއްޗެއްހެއްޔ؟ ހުރިހާ އެއްޗަކީ މާއްދާކަމަށް ބުނުމަކީ އިންސާނާގެ ހަށިގަނޑުގެ ހުރިހާ ބައެއްވެސް ފުރަމޭ އަދި ހިތް އަދި ގުރުދާކަމަށް ބުނުންފަދަ އިމްޕީރިކަލް އަދި މޮޔަ ބުނުމެކެވެ. ޔަޤީނުންވެސް އެއް އެއްޗަކީ އެއް އެއްޗެވެ އަދި އަނެއް އެއްޗަކީ އަނެއް އެއްޗެވެ، ކޮންމެ ގުނަވަނަކީ ތަފާތު އެއްޗެކެވެ އަދި ކޮންމެ މާއްދާއަކީވެސް ތަފާތު އެއްޗެކެވެ. ދެން މި ހުރިހާ މާއްދާތަކުގެ ތެރެއިން މި އެންމެ މަޝްހޫރު މާއްދާއަކީ ކޮބައިހެއްޔ؟

ޕެންޑިއުލަމްގެ ކޮންސެޕްޓްތަކާއި އެކު އެތައް ބަޔަކު ކުޅެއެވެ، އެހެންނަމަވެސް ހަޤީޤަތުގައި ކޮންސެޕްޓްތަކަކީ ހަޤީޤަތެއް ނޫނެވެ.

ސިކުނޑިއަށް އެނގެނީ ހަމައެކަނި ތަބީޢަތުގެ ހިޔާލީ ސިފަތަކެވެ، އެހެންނަމަވެސް އެފަދަ ސިފަތަކުގައި ހިމެނޭ ހަޤީޤަތަކާ ބެހޭގޮތުން އެއްވެސް އެއްޗެއް ނޭނގެއެވެ.

ތިއަރީތައް ވަގުތާއި އަހަރުތަކާއެކު ފަހުގެ ފެޝަނަށް ބަދަލުވެއެވެ، އަދި ސްކޫލުން ދަސްވި އެއްޗެއް ފަހުން ބޭކާރުވެއެވެ؛ ނަތީޖާއަކީ: އެއްވެސް މީހަކަށް އެއްވެސް އެއްޗެއް ނޭނގެއެވެ.

ޕެންޑިއުލަމްގެ އެންމެ ކަނާތްފަރާތުގެ ނުވަތަ އެންމެ ވާތްފަރާތުގެ ކޮންސެޕްޓްތައްވެސް އަންހެނުންގެ ފެޝަންތައް ފަދައިން ފާއިތުވެގެންދެއެވެ. އެ ހުރިހާ ކަމަކީ ސިކުނޑީގެ ޕްރޮސެސްތަކާއި ވިސްނުމުގެ ބޭރުފުށުގައި ހިނގާ ކަންތައްތަކާއި އަދި ބުއްދީގެ ކުޑަކަމުދާ ކަންތައްތަކާއި ބޮޑާކަންތަކެވެ.

ކޮންމެ ސައިކޮލޮޖިކަލް ޑިސިޕްލިންއަކާ ދެކޮޅަށްވެސް އެހެން ޑިސިޕްލިންއެއް އުފެދިގެންދެއެވެ، މަންޠިޤީ ގޮތުން އެކުލަވާލާފައިވާ ކޮންމެ ސައިކޮލޮޖިކަލް ޕްރޮސެސްއަކާ ދެކޮޅަށްވެސް އެފަދަ އެހެން ޕްރޮސެސްއެއް އުފެދިގެންދެއެވެ، އަދި ހުރިހާކަމެއް ނިމުނުއިރު، ކޮބައިހޭ؟

ހަޤީޤީ، ތެދު، އެއީ އަހަރެމެންގެ ޝައުޤުވެރިކަން ހުރި އެއްޗެވެ؛ ނަމަވެސް މިއީ ޕެންޑިއުލަމްގެ މައްސަލައެއް ނޫނެވެ، ތިއަރީތަކާއި އީމާންކަމުގެ އޮއިވަރުގެ ތެރެއަކު އެތަނެއް ނުހުރެއެވެ.

ހަޤީޤަތަކީ ކޮންމެ ވަގުތަކު، ކޮންމެ ހިނދަކު ނޭނގޭ އެއްޗެވެ.

ހަޤީޤަތްވަނީ ޕެންޑިއުލަމްގެ މެދުގައެވެ، އެންމެ ކަނާތްފަރާތުގައެއް ނޫނެވެ އަދި އެންމެ ވާތްފަރާތުގައެއްވެސް ނޫނެވެ.

އީސާގެފާނު ކިބައިން ސުވާލުކުރެވުނުހިނދު: ހަޤީޤަތަކީ ކޮބައިހެއްޔ؟، އެކަލޭގެފާނު ފުން ހަމަހިމޭންކަމެއް އިޚްތިޔާރުކުރެއްވިއެވެ. އަދި ހަމަ އެ ސުވާލު ބުއްދާގެފާނު ކިބައިން ކުރެވުނުހިނދު، އެކަލޭގެފާނު ފުރަގަސްދެއްވާފައި އެތަނުން ވަޑައިގެންފިއެވެ.

ހަޤީޤަތަކީ ޚިޔާލުތަކާއި، ތިއަރީތަކާއި، އެންމެ ކަނާތްފަރާތުގެ ނުވަތަ އެންމެ ވާތްފަރާތުގެ ޕްރިޖުޑިސްތަކުގެ މައްސަލައެއް ނޫނެވެ.

ސިކުނޑިން ހަދާ ކޮންސެޕްޓަކީ ދުވަހަކުވެސް ހަޤީޤަތަކަށް ނުވާނެއެވެ.

ހަޤީޤަތާ ބެހޭގޮތުން ވިސްނުމުގައިވާ ޚިޔާލަކީ ދުވަހަކުވެސް ހަޤީޤަތަކަށް ނުވާނެއެވެ.

ތިމާގެ ނަޒަރުގައި ހަޤީޤަތާ މެދު އޮންނަ އެންމެ މާތް ޚިޔާލަކީވެސް، އެއްގޮތަކަށްވެސް ހަޤީޤަތެއް ނޫނެވެ.

ރޫހާނީ ޖަރަޔާންތަކާއި މާއްދީ ޖަރަޔާންތަކަށް ދުވަހަކުވެސް ހަޤީޤަތަށް އަހަރެމެން ގެންދެވޭކަށް ނެތެވެ.

ހަޤީޤަތަކީ ސީދާގޮތުން ތަޖުރިބާކުރަންޖެހޭ އެއްޗެކެވެ، އަލިފާނަށް އިނގިލި ވައްދާލުމުން އަނދާ ފަދައިން ނުވަތަ ފެންބޮއެގެން މަރުވާފަދައިންނެވެ.

ޕެންޑިއުލަމްގެ މެދު އަހަރެމެންގެ ތެރޭގައި އެބަހުއްޓެވެ، އަދި އެތަނުގައި އަހަރެމެން ސީދާގޮތުން ހޯދަންޖެހޭނީ ހަޤީޤީ އެއްޗެކެވެ، ތެދު އެއްޗެކެވެ.

އަމިއްލަ ނަފްސު ހޯދާ، އަމިއްލަ ނަފްސު ދެނެގަތުމަށްޓަކައި އަދި އަމިއްލަ ނަފްސު ފުންކޮށް ދެނެގަތުމަށްޓަކައި ސީދާގޮތުން އަމިއްލަ ނަފްސު ހޯދަން ޖެހެއެވެ.

ހަޤީޤަތުގެ ތަޖުރިބާ ލިބިގެންދަނީ ހަމައެކަނި އިންސާނާގެ ހުރިހާ އެސެންޝިއަލް ބައިތަކެއް ނައްތާލީމައެވެ.

ހަމައެކަނި ގޯސްކަން ނައްތާލީމާ ތެދުކަން ލިބިގެންދެއެވެ. ހަމައެކަނި "އަމިއްލަ ނަފްސު"، ތިމާގެ ގޯސްތައް، ތިމާގެ ޕްރިޖުޑިސްތައް އަދި ބިރުތައް، ތިމާގެ ޝައުޤުވެރިކަން އަދި އެދުންތައް، އީމާންކަން އަދި ފާފަތައް، ބުއްދީގެ ވަށާކުރުމާއި ހުރިހާ ވައްތަރެއްގެ އަމިއްލަ ފުދުންތެރިކަން ނައްތާލީމާ ހަޤީޤީ އެއްޗެއްގެ ތަޖުރިބާ އަހަރެމެންނަށް ލިބިގެންދެއެވެ.

ހަޤީޤަތަށް ބުނެވުނު ނުވަތަ ނުބުނެވުނު، ލިޔެވުނު ނުވަތަ ނުލިޔެވުނު އެއްޗަކާ އެއްވެސް ގުޅުމެއް ނެތެވެ، ހަޤީޤަތް އަހަރެމެންނަށް ލިބިގެންދަނީ ހަމައެކަނި "އަމިއްލަ ނަފްސު" މަރުވީމައެވެ.

ސިކުނޑިއަށް ހަޤީޤަތް ހޯދާކަށް ނޭނގޭނެ، އެހެނީ އެއަށް އެނގޭ އެއްޗެއް ނެތެވެ. ސިކުނޑިއަށް ހަޤީޤަތް ދެނެގަންނާކަށް ނޭނގޭނެ، އެހެނީ އެއަށް ދުވަހަކުވެސް އެނގިފައެއް ނުވެއެވެ. ހަޤީޤަތް އަހަރެމެންނަށް އަމިއްލައަށް ލިބިގެންދަނީ ހަމައެކަނި "އަމިއްލަ ނަފްސު" އުފައްދާ ނުރަނގަޅު އެލިމެންޓްތައް ނައްތާލީމައެވެ.

ސަމާލުކަން އަދިވެސް އަމިއްލަ ނަފްސުގައި ބަންދުވެފައި އޮތްހާ ހިނދަކު ހަޤީޤީކަމާއި ހަށިގަނޑުން، ލޯބިން އަދި ސިކުނޑިން މާ މައްޗަށްވާ ހަޤީޤަތުގެ ތަޖުރިބާއެއް ނުކުރެވޭނެއެވެ.

އަމިއްލަ ނަފްސު ކާއިނާތުގެ އަޅިއަށް ބަދަލުވުމުން، ހޭލުންތެރިވުން ސަލާމަތްވެގެންދަނީ ދާއިމީ ގޮތަކަށް ހޭލައި ސީދާގޮތުން ހަޤީޤަތުގެ ތަޖުރިބާކުރުމަށްޓަކައެވެ.

ބޮޑު ކަބީރު އީސާގެފާނު ހަމަހޭލާ ހުންނަވައި ބަސްފުޅު ވިދާޅުވިއެވެ: "ތެދު ދެނެގަންނާށެވެ، އަދި އޭތި ތިޔަބައިމީހުން މިނިވަންކޮށްދޭނެއެވެ."

އިންސާނަކަށް ފަންސާސްހާސް ތިއަރީ ދަސްވުމުން ކޮށްދޭނީ ކޮން ފައިދާއެއްހެއްޔ؟، އޭނާ ދުވަހަކުވެސް ހަޤީޤަތުގެ ތަޖުރިބާކޮށްފައިނުވާނަމަ؟

އިންސާނަކު ބުއްދީގެ ނިޒާމަކީ ވަރަށް މާތް އެއްޗެކެވެ، އެހެންނަމަވެސް ކޮންމެ ނިޒާމަކާ ދެކޮޅަށްވެސް އެހެން ނިޒާމެއް އުފެދިގެންދެއެވެ، އެއިން އެއްވެސް އެއްޗަކީ ހަޤީޤަތެއް ނޫނެވެ.

އެހެންކަމުން އެންމެ ރަނގަޅީ އަމިއްލަ ނަފްސު ހޯދާ އަމިއްލަ ނަފްސު ދެނެގަތުމަށް މަސައްކަތްކުރުމެވެ އަދި އެއްދުވަހުން ސީދާގޮތުން ހަޤީޤީއެއްޗެއް، ހަޤީޤަތެއް ތަޖުރިބާކުރުމެވެ.
