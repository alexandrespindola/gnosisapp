ދެލްފޫސްގެ ފައްޅީގެ މަތިވެރި ދޮރުކައިރީގައި ދިރިހުރި ގަލުގައި ކަނޑާފައިވާ ލިޔުމެއް ވެއެވެ. "ނޯ ސެ ޓެ އިޕްސަމް". ތިބާގެ އަމިއްލަ ނަފްސު ދެނެގަންނާށެވެ. އޭރުން ތިބާއަށް ކާއިނާތާއި އިލާހުން އެނގިވަޑައިގަންނަވާނެއެވެ.

މެޑިޓޭޝަންގެ މަތިވެރި ސައިންސްގެ މައިގަނޑު އަސާސަކީ ގަދީމީ ހައިރޯފަންޓްސް ގްރީކްސްގެ މި މުޤައްދަސް ޝިއާރެވެ.

އަހަރެމެން ހަޤީޤަތުގައިވެސް އަދި ވަރަށް އިޚްލާޞްތެރިކަމާއެކު މެޑިޓޭޝަންގެ ރަނގަޅު އަސާސެއް ޤާއިމުކުރަން ބޭނުންނަމަ، އެންމެ މުހިއްމީ ކޮންމެ މީހަކުވެސް އެމީހެއްގެ އަމިއްލަ ނަފްސު މަންފާގެ ހުރިހާ ފެންވަރަކުންވެސް ދެނެގަތުމެވެ.

މެޑިޓޭޝަންގެ ރަނގަޅު އަސާސެއް ޤާއިމުކުރުމަކީ ހަޤީޤަތުގައިވެސް އެދުންވެރިކަމާއި، އަމިއްލަ އެދުމާއި، ބިރުވެރިކަމާއި، ނަފްރަތާއި، ނަފްސާނީ ބާރުތައް ހޯދުމުގެ ދަހާއި، ނަތީޖާއަށް އެދުންވެރިވުމާއި، މިފަދަ ކަންކަމުން ސަލާމަތްވުމެވެ.

މެޑިޓޭޝަންގެ މައިގަނޑު އަސާސް ޤާއިމުކުރުމަށްފަހު ސިކުނޑި ހަމަހިމޭންވެ، ފުން އަދި މަތިވެރި ހަމަހިމޭންކަމެއްގައި ވާކަން އެންމެންނަށްވެސް ސާފުވާނެއެވެ.

ރަނގަޅު ލޮޖިކްގެ ނަޒަރުން ބަލާއިރު، އަމިއްލަ ނަފްސު ނޭނގި ހަޤީޤަތް ތަޖުރިބާކުރަން ބޭނުންވުމަކީ މޮޔަކަމެކެވެ.

ސިކުނޑީގެ ކޮންމެ މައިދާނަކުންވެސް، ކޮންމެ މައްސަލައަކާއި، ކޮންމެ އެދުމަކާއި، ކޮންމެ ހަނދާނަކާއި، ކޮންމެ ނަފްސާނީ އުނިކަމެއް ފަދަ ކޮންމެ ކަމެއް އިންޓެގްރަލް ގޮތަކަށް ދެނެގަތުން ވަރަށް މުހިއްމެވެ.

މެޑިޓޭޝަންގެ ތެރޭގައި އަހަރެމެންގެ ސިކުނޑީގެ ސްކްރީނުން އަހަރެމެންގެ ސިފަތަކުގައި ހިމެނޭ ނަފްސާނީ އުނިކަންތަކާއި، އުފާވެރިކަމާއި، ހިތާމަތަކާއި، ގިނަ އަދަދެއްގެ ހަނދާންތަކާއި، އެތެރެއިންނާއި ބޭރުން އަންނަ އެތައް އިންޕަލްސްތަކާއި، ހުރިހާ ވައްތަރެއްގެ އެދުންތަކާއި، އެކި ބާވަތުގެ ޝައުޤުވެރިކަންތަކާއި، ދުވަސްވީ ރުޅިވެރިކަންތަކާއި، ނަފްރަތުތައް ފަދަ ކަންކަން ފާސްވެގެންދާކަން އެންމެންނަށްވެސް ސާފުވާނެއެވެ.

ހަޤީޤަތުގައިވެސް މެޑިޓޭޝަންގެ އަސާސް އޭނާގެ ސިކުނޑީގައި ޤާއިމުކުރަން ބޭނުންވާ މީހާ، އަހަރެމެންގެ ވިސްނުމުގެ ރަނގަޅު އަދި ނުބައި އަގުތަކަށް ފުރިހަމަ ސަމާލުކަމެއް ދީ، ހަމައެކަނި އިލްމީ ފެންވަރުގައި އެކަންކަން ދެނެގަތުމުގެ ބަދަލުގައި ސިކުނޑީގެ ސަބްކޮންޝަސް، އިންފްރާކޮންޝަސް އަދި އަންކޮންޝަސްގެ ހުރިހާ މައިދާނެއްގައިވެސް އެކަންކަން ފުރިހަމަގޮތަކަށް ދެނެގަންނަންވާނެއެވެ. އަހަރެމެން ދުވަހަކުވެސް ހަނދާން ނައްތާލައިގެން ނުވާނެއެވެ. ސިކުނޑިއަށް އެތައް ފެންވަރެއް ލިބިގެންވެއެވެ.

މި ހުރިހާ އަގުތަކުގެ ފުން ދިރާސާއަކީ ހަޤީޤަތުގައިވެސް އަމިއްލަ ނަފްސު ދެނެގަތުމެވެ.

ސިކުނޑީގެ ސްކްރީނުގައި އުފެދޭ ކޮންމެ ފިލްމެއްގެވެސް ފެށުމަކާއި ނިމުމެއް ވެއެވެ. ޝަޚްސުތަކާއި، އެދުންތަކާއި، ޝައުޤުވެރިކަންތަކާއި، އެމްބިޝަންތަކާއި، ހަނދާންތަކުގެ ޕެރޭޑް ނިމުމުން ސިކުނޑި ހަމަހިމޭންވެ، ހުރިހާ ވައްތަރެއްގެ ޚިޔާލުތަކުން ހުސްވެގެން ފުން ހަމަހިމޭންކަމެއްގައި ވެއެވެ.

ސައިކޮލޮޖީ ދާއިރާއިން މިހާރު ތަޢުލީމު ޙާޞިލުކުރަމުންދާ ދަރިވަރުން އިލްމީ ހުސްކަން ތަޖުރިބާކުރަން ޖެހެއެވެ. އަހަރެމެންގެ އަމިއްލަ ސިކުނޑީގެ ތެރެއަށް ހުސްކަން ވަނުމުން ބަދަލުކުރާ އެލިމެންޓެއް ތަޖުރިބާކޮށް، އިޙްސާސްކޮށް، ދިރިއުޅެވޭނެއެވެ. އެއީ ހަޤީޤަތެވެ.

ހަމަހިމޭންކަމުގައިވާ ސިކުނޑިއަކާއި ގަދަކަމުން ހަމަހިމޭންކުރެވިފައިވާ ސިކުނޑިއެއް ވަކިކުރާށެވެ.

ހަމަހިމޭންކަމުގައިވާ ސިކުނޑިއަކާއި ގަދަކަމުން ހަމަހިމޭންކުރެވިފައިވާ ސިކުނޑިއެއް ވަކިކުރާށެވެ.

ކޮންމެ ލޮޖިކަލް ނަތީޖާއެއްގެ އަލީގައި އަހަރެމެން ދެނެގަންނަން ޖެހެނީ ސިކުނޑި ގަދަކަމުން ހަމަހިމޭންކުރެވިފައިވާ ހާލަތެއްގައި ފުންމިނުގައި އަދި އެހެން ފެންވަރުތަކުގައި އެ ސިކުނޑި ހަމަހިމޭނުން ނުހުންނާނެއެވެ. އަދި މިނިވަންވުމަށްޓަކައި މަސައްކަތްކުރާނެއެވެ.

އެނަލިޓިކަލް ނަޒަރުން ބަލާއިރު، ސިކުނޑި ގަދަކަމުން ހަމަހިމޭންކުރެވިފައިވާ ހާލަތެއްގައި އެ ސިކުނޑި ފުންމިނުގައި ހަމަހިމޭނުން ނުހުންނާނެއެވެ. ހަޅޭއްލަވައި، ވަރަށް ބޮޑަށް މާޔޫސްވާނެއެވެ.

ސިކުނޑީގެ ހަޤީޤީ ހަމަހިމޭންކަމާއި، އަމިއްލައަށް އުފެދޭ ހަމަހިމޭންކަން އަހަރެމެންނަށް އަންނަނީ ރަޙްމަތެއްގެ ގޮތުގައެވެ. އުފާވެރިކަމެއްގެ ގޮތުގައެވެ. އަހަރެމެންގެ އަމިއްލަ ވުޖޫދުގެ ވަރަށް ޚާއްޞަ ފިލްމެއް ބުއްދީގެ މޮޅު ސްކްރީނުގައި ނިމުމުންނެވެ.

ސިކުނޑި އަމިއްލައަށް އުފެދިގެން ހަމަހިމޭންވީމާ އެކަންޏެވެ. ސިކުނޑި މީރު ހަމަހިމޭންކަމެއްގައި ވީމާ އެކަންޏެވެ. އިލްމީ ހުސްކަމުގެ ބާރުގަދަ އިރުއެރުން އަންނާނީއެވެ.

ހުސްކަން ބަޔާންކުރަން ފަސޭހައެއް ނޫނެވެ. މާނަކުރެވޭކަށެއް ނެތެވެ. ނުވަތަ ސިފަކުރެވޭކަށެއް ނެތެވެ. އެއާ ބެހޭގޮތުން އަހަރެމެން ކުރާ ކޮންމެ ޚިޔާލެއްވެސް މައިގަނޑު ނުކުތާއަށް ފެއިލްވެދާނެއެވެ.

ހުސްކަން ސިފައެއް ނުކުރެވޭނެއެވެ. ނުވަތަ ލަފުޒުތަކުން ބަޔާނެއް ނުކުރެވޭނެއެވެ. އެއީ މައިގަނޑު ގޮތެއްގައި އިންސާނުންގެ ބަސް އުފައްދާފައިވަނީ ވުޖޫދުގައި ހުރި އެއްޗެހި، ޚިޔާލުތައް އަދި އިޙްސާސްތައް ފާހަގަކުރުމަށްޓަކައެވެ. ވުޖޫދުގައި ނުވާ ކަންކަމާއި، އެއްޗެހި އަދި އިޙްސާސްތައް ސާފު އަދި ޚާއްޞަ ގޮތަކަށް ބަޔާންކުރުމަށް އެ ބަސް އެކަށީގެންވާ ނުވާތީއެވެ.

ވުޖޫދުގެ ސިފަތަކުން ހަނިކުރެވިފައިވާ ބަހެއްގެ ތެރޭގައި ހުސްކަމާއި މެދު ވާހަކަދައްކަން މަސައްކަތްކުރުމަކީ ޝައްކެއްވެސް ނެތް ގޮތުގައި ހަޤީޤަތުގައިވެސް މޮޔަކަމެކެވެ. އަދި މުޅިންވެސް ގޯހެކެވެ.

"ހުސްކަމަކީ ނޯ-އެގްޒިސްޓެންސްއެވެ. އަދި އެގްޒިސްޓެންސްއަކީ ހުސްކަމެއް ނޫނެވެ."

"ސިފައެއް ހުސްކަމަކާ ތަފާތެއް ނުވެއެވެ. އަދި ހުސްކަމެއްވެސް ސިފައަކާ ތަފާތެއް ނުވެއެވެ."

"ސިފައަކީ ހުސްކަމެވެ. އަދި ހުސްކަމަކީ ސިފައެކެވެ. ކަންތައްތައް ވުޖޫދުގައި ހުންނަނީ ހުސްކަމުގެ ސަބަބުންނެވެ."

"ހުސްކަމާއި އެގްޒިސްޓެންސް އެކަކު އަނެކަކަށް ފުރިހަމަކޮށްދެއެވެ. އަދި ދެކޮޅުވެރިވެސް ނުވެއެވެ." ހުސްކަމާއި އެގްޒިސްޓެންސް އެކުގައި ހިމެނި، ގައިގޯޅިވެއެވެ.

"ސެންސިބިލިޓީ ނޯމަލް މީހުން އެއްޗަކު ދެކޭއިރު، އެމީހުންނަށް އެ ފެންނަނީ އެ އެތީގެ ވުޖޫދީ މަންޒަރެވެ. ހުސްކަމުގެ މަންޒަރެއް އެމީހުންނަކަށް ނުފެނެއެވެ."

"ހުރިހާ އެންލައިޓެންޑް އިންސާނުންނަށް ކޮންމެ އެއްޗެއްގެވެސް ވުޖޫދީ އަދި ހުސްކަމުގެ މަންޒަރު އެއްފަހަރާ ފެނިދާނެއެވެ."

"ހުސްކަމަކީ ހަމައެކަނި މާއްދީނުވާ އަދި ޒާތީނުވާ ނިސްބަތް ދައްކުވައިދޭ ލަފުޒެކެވެ. އަދި އެއީ މުޅިންވެސް ވަކިވެ، މިނިވަންވެފައިވާ ޙާލަތުގެ އިޝާރާތެކެވެ."

ސްކޫލްތަކާއި، ކޮލެޖްތަކާއި، ޔުނިވަރސިޓީތަކުގެ މުދައްރިސުން އަހަރެމެންގެ އިންޤިލާބީ ސައިކޮލޮޖީ ފުންކޮށް ދިރާސާކޮށް، އެއަށްފަހު އެމީހުންގެ ދަރިވަރުންނަށް ހަޤީޤަތް ތަޖުރިބާކުރުމަށް ގެންދާ މަގު އުނގަންނަވައިދޭންވާނެއެވެ.

ޚިޔާލު ނިމުމަކަށް އައުމުން ހަޤީޤަތުގެ ތަޖުރިބާއަށް ވާސިލުވުމަކީ ހަމައެކަނި ވެވެން އޮތް ކަމެކެވެ.

ހުސްކަން ބާރުގަދަ އިރުއެރުމެއްގެ ސަބަބުން އަހަރެމެންނަށް ޕިއުރަ ރިއަލިޓީގެ ސާފު އަލި ތަޖުރިބާކުރެވޭނެއެވެ.

ސިފައެއްނެތް އަދި ކުލައެއްވެސްނެތް ހަޤީޤަތުގައި އެ ވަގުތު ހުރި އެނގުންވެރިވުން ހުސްވެފައިވާއިރު، ތަބީޢަތުގެ ހުސްކަމާއެކު، އެއީ ހަޤީޤީ ރިއަލިޓީއެވެ. ޔުނިވަރސަލް ހެވެވެ.

ތިބާގެ ބުއްދީގެ ހަޤީޤީ ތަބީޢަތަކީ ނަތިންގްގެ ހުސްކަމެއްގެ ގޮތުގައި ނުބަލަންވާނެ ހުސްކަމެވެ. އެއީ ހުރަހެއްނެތި ބުއްދީ އަމިއްލައެވެ. އަލިގަދަ، ޔުނިވަރސަލް އަދި އުފާވެރި އެއީ ޚަބަރުދާރުވުމެވެ. ޔުނިވަރސަލް ގޮތެއްގައި ދަންނަ ބުއްދާއެވެ.

ތިބާގެ އަމިއްލަ ހުސް ޚަބަރުދާރުވުމާއި އަލިގަދަ އަދި އުފާވެރި ބުއްދީ ވަކިނުކުރެވޭނެއެވެ. އޭގެ ޔުނިއަން އަކީ ދަރުމާ-ކާޔާއެވެ. ފުރިހަމަ އިލްމުވެރިވުމުގެ ޙާލަތެވެ.

ތިބާގެ އަމިއްލަ އަލިގަދަ ޚަބަރުދާރުވުމާއި، ހުސްކަމާއި، އިސްޕްލެންޑަރގެ ބޮޑު ހަށިގަނޑުން ވަކިނުކުރެވޭނެއެވެ. އުފަންވުމެއް އަދި މަރުވުމެއްވެސް ނެތެވެ. އަދި އެއީ ބަދަލުނުވާ އަލި އަމިތާރާ ބުއްދާއެވެ.

މި އެނގުންވެރިވުން ފުދެއެވެ. ތިބާގެ އަމިއްލަ ބުއްދީގެ ހުސްކަން ބުއްދާގެ ޙާލަތުގެ ގޮތުގައި ދެނެގަތުމާއި ތިބާގެ އަމިއްލަ ޚަބަރުދާރުވުމުގެ ގޮތުގައި ގަބޫލުކުރުމަކީ ބުއްދާގެ ރޫޙާނީ ރަސްކަމުގައި ކުރިއަށް ދިއުމެވެ.

މެޑިޓޭޝަންގެ ތެރޭގައި ތިބާގެ އިންޓެލެކްޓް ޑިސްޓްރެކްޓް ނުވެ، ރައްކާކުރާށެވެ. ތިޔަބޭފުޅުން މެޑިޓޭޝަންގައި ތިއްބެވިކަން ހަނދާން ނައްތާލާށެވެ. ތިޔަބޭފުޅުން މެޑިޓޭޓްކުރައްވާކަމަށް ހީނުކުރައްވާށެވެ. އެހެނީ މެޑިޓޭޓްކުރާކަމަށް ޚިޔާލުކުރާ ހިނދުވެސް އެ ޚިޔާލު މެޑިޓޭޝަންއަށް އުނދަގޫކުރަން ފުދެއެވެ. ހަޤީޤަތް ތަޖުރިބާކުރުމަށް ތިބާގެ ސިކުނޑި ހުސްވާން ޖެހޭނެއެވެ.
