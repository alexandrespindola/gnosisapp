འཚོ་བའི་ཉམས་ལེན་གྱི་ཁྱབ་ཁོངས་ནང་ང་ཚོས་ནམ་ཡང་ཧ་ལས་པའི་འགལ་འདུ་རྙེད་ཐུབ། རྒྱུ་དངུལ་མང་པོ་ཡོད་པའི་ཁྱིམ་ཚང་དང་གྲོགས་པོ་མང་པོ་ཡོད་མཁན་ཚོས་སྐབས་རེ་ཧ་ཅང་སྡུག་བསྔལ་མྱོང་གི་རེད། འཇོར་དང་གོག་བརྡུག་རྒྱག་མཁན་གྱི་ངལ་རྩོལ་བ་དམན་པ་ཚོ་དང་ཡང་ན་འབྱོར་འབྲིང་གྲལ་རིམ་གྱི་མི་ཚོས་སྐབས་རེ་སྐྱིད་པོ་འཚོ་བ་སྐྱེལ་གི་རེད།

དངུལ་ཐེར་འབུམ་མང་པོ་ཡོད་མཁན་མང་པོ་ཞིག་ལ་འཁྱིག་སྟོབས་མེད་པའི་ནད་དང་ཨ་མ་ཕྱུག་མོ་ཚོས་རང་གི་ཁྱོ་གའི་བློ་སེམས་འགྱུར་བའི་རྐྱེན་གྱིས་སྐྱོ་བས་མིག་ཆུ་གཏོང་གི་རེད། འཛམ་བུའི་གླིང་འདིའི་ཕྱུག་བདག་ཚོ་གསེར་གྱི་ཁྱིམ་ཚང་ནང་གི་རྒོད་པོ་དང་འདྲ་བར་དུས་ཚོད་འདི་ཚོའི་ནང་སྲུང་དམག་མེད་ན་འཚོ་ཐུབ་ཀྱི་མ་རེད། རྒྱལ་ཁབ་ཀྱི་མི་དྲག་ཚོར་ལྕགས་སྒྲོག་བརྒྱབ་ཡོད་པ་དང་འདྲ་བར་རང་དབང་གཏན་ནས་མེད་པ་དང་མཚོན་ཆ་བསྣམས་པའི་མི་ཚོས་མཐའ་སྐོར་ནས་གང་སར་འགྲོ་དགོས་ཀྱི་རེད།

ང་ཚོས་གནས་སྟངས་འདི་ལ་ཞིབ་མོར་སློབ་སྦྱོང་བྱེད་དགོས་ཀྱི་འདུག ང་ཚོས་འཚོ་བ་ཟེར་བ་ག་རེ་ཡིན་མིན་ཤེས་དགོས་ཀྱི་འདུག མི་རེ་ངོ་རེར་རང་གི་འདོད་མོས་ལྟར་བསམ་ཚུལ་བརྗོད་པའི་རང་དབང་ཡོད། གང་བཤད་ཀྱང་བཤད་མཁན་ཚོས་གཏན་ནས་ཤེས་ཀྱི་མ་རེད་ལ་འཚོ་བ་ནི་མི་སུ་གཅིག་གིས་ཀྱང་ཧ་མི་གོ་བའི་དཀའ་རྙོག་ཅིག་རེད།

མི་ཚོས་རང་གི་མི་ཚེའི་ལོ་རྒྱུས་སོགས་སྟབས་བདེ་པོ་ཡོང་ཆེད་དུ་དོན་རྐྱེན་དང་མིང་རུས་དང་དུས་ཚོད་སོགས་བརྗོད་ཀྱི་ཡོད་པ་དང་དེ་འདྲའི་སྒྲུང་བཤད་དུས་དགའ་ཚོར་སྐྱེ་གི་རེད། མི་དབུལ་པོ་དེ་ཚོས་རང་གི་སྒྲུང་དེ་ཆ་ཚང་བ་མིན་པ་ཤེས་ཀྱི་མེད་པ་དང་དོན་རྐྱེན་དང་མིང་དང་དུས་ཚོད་སོགས་ནི་གློག་བརྙན་གྱི་ཕྱི་ཚུགས་ཁོ་ན་ཡིན་པ་དང་ནང་གི་གནས་སྟངས་ཆད་ཡོད་པ་རེད།

རྣམ་རྟོག་གི་གནས་སྟངས་ཤེས་རྒྱུ་ཧ་ཅང་གལ་ཆེན་པོ་ཡིན་པ་དང་དོན་རྐྱེན་རེ་རེ་བཞིན་དེ་དང་འབྲེལ་བ་ཡོད་པའི་སེམས་ཀྱི་གནས་སྟངས་རེ་ཡོད་པ་རེད། གནས་སྟངས་དེ་ནང་ཡོད་པ་དང་དོན་རྐྱེན་དེ་ཕྱི་རུ་ཡོད་པ་དང་ཕྱིའི་དོན་རྐྱེན་རྣམས་ནི་ཚང་མ་མིན།

ནང་གི་གནས་སྟངས་ནི་བསམ་པ་བཟང་ངན་དང་སེམས་ཁྲལ་དང་སྐྱོ་བ་དང་ལྷ་འདྲེ་ལ་ཡིད་ཆེས་དང་ཞེད་སྣང་དང་དོགས་པ་དང་སྙིང་རྗེ་དང་རང་གིས་རང་ལ་བརྩི་འཇོག་དང་རང་ཉིད་ཆེན་པོར་བརྩི་བ་དང་དགའ་བའི་ཚོར་བ་དང་སྐྱིད་པའི་ཚོར་བ་སོགས་སོགས་ཡིན་པ་ཤེས་དགོས།

ནང་གི་གནས་སྟངས་རྣམས་ཕྱིའི་དོན་རྐྱེན་དང་འབྲེལ་བ་ཡོད་པའམ་ཡང་ན་དེ་དག་ལས་བྱུང་བ་ཡིན་པ་དང་ཡང་ན་དེ་དག་དང་འབྲེལ་བ་མེད་པ་ཡོང་སྲིད་པ་ནི་བསྙོན་དུ་མེད་པ་ཞིག་རེད། གང་ལྟར་ཡང་གནས་སྟངས་དང་དོན་རྐྱེན་གཉིས་ខྱ་ཁྱལ་བ་རེད། དོན་རྐྱེན་རྣམས་ནམ་ཡང་གནས་སྟངས་དང་མཐུན་པ་མིན།

དགའ་འོས་པའི་དོན་རྐྱེན་ཞིག་གི་ནང་གི་གནས་སྟངས་དེའི་དོན་རྐྱེན་དང་མཐུན་པ་མེད་སྲིད། སྡུག་བསྔལ་གྱི་དོན་རྐྱེན་ཞིག་གི་ནང་གི་གནས་སྟངས་དེའི་དོན་རྐྱེན་དང་མཐུན་པ་མེད་སྲིད། ང་ཚོས་དུས་ཡུན་རིང་པོར་སྒུག་ཡོད་པའི་དོན་རྐྱེན་བྱུང་དུས་ང་ཚོས་ག་རེ་ཞིག་ཆད་པ་འདྲ་ཞིག་ཚོར་གྱི་རེད།

ངེས་པར་དུ་ཕྱིའི་དོན་རྐྱེན་དང་མཉམ་དུ་འདྲེས་དགོས་པའི་ནང་གི་གནས་སྟངས་དེ་ཆད་པ་རེད། མང་པོ་ཞིག་ལ་ང་ཚོས་སྒུག་མེད་པའི་དོན་རྐྱེན་དེ་བྱུང་བ་དེ་གནས་སྐབས་བཟང་ཤོས་སྦྱིན་མཁན་དེ་ཆགས་ཡོང་གི་རེད།
