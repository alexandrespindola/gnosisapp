ལྟ་ཞིབ་བྱེད་སྐབས་དོ་སྣང་ཕྱི་རོལ་ཏུ་ཁ་ཕྱོགས་ཏེ་ཕྱིའི་འཇིག་རྟེན་ལ་ཚོར་བའི་སྒོ་ཁུང་བརྒྱུད་ནས་ཕྱོགས་པ་ཡིན།

རང་ཉིད་ལ་རང་གིས་ལྟ་ཞིབ་བྱེད་སྐབས་དོ་སྣང་ནང་དུ་ཁ་ཕྱོགས་ཤིང་དེའི་ཆེད་ཕྱིའི་ཚོར་བའི་དབང་པོ་དག་གིས་ཕན་ཐོགས་མི་ཐུབ་པས། རྒྱུ་མཚན་དེ་ནི་གསར་ཞུགས་པས་རང་གི་སེམས་ཁམས་ཀྱི་ནང་གི་བརྒྱུད་རིམ་ལ་ལྟ་ཞིབ་བྱེད་དཀའ་བའི་རྒྱུ་མཚན་འདང་ངེས་ཤིག་ཡིན།

ཚན་རིག་གཞུང་འབྲེལ་གྱི་ལག་ལེན་གྱི་ཕྱོགས་ཀྱི་འགོ་འཛུགས་ས་ནི་བལྟ་ཞིབ་བྱེད་ཆོག་པ་ཞིག་ཡིན། རང་ཉིད་ཐོག་ལས་ཀ་བྱེད་པའི་འགོ་འཛུགས་ས་ནི་རང་ལ་ལྟ་ཞིབ་བྱེད་པ་སྟེ་རང་ལ་བལྟ་ཞིབ་བྱེད་ཆོག་པ་ཞིག་ཡིན།

གོང་དུ་བཀོད་པའི་འགོ་འཛུགས་ས་གཉིས་པོ་འདིས་ང་ཚོ་ཕྱོགས་མི་འདྲ་བའི་ནང་དུ་ཁྲིད་འགྲོ་བ་ནི་བསྙོན་དུ་མེད་པ་ཞིག་ཡིན།

ཚན་རིག་གཞུང་འབྲེལ་གྱི་བཟོད་སྒོམ་རིང་ལུགས་ཀྱི་བར་དུ་མགོ་འཁོར་ནས་ཕྱིའི་སྣང་ཚུལ་སློབ་སྦྱོང་བྱེད་པ་དང་། ཕྲ་ཕུང་དང་། རྡུལ་ཕྲན་དང་། ཆ་རྡུལ་དང་། ཉི་མ་དང་། སྐར་མ་དང་། འཕྲུལ་སྐར་སོགས་ལ་བལྟ་ཞིབ་བྱེད་པའི་ནང་དུ་ཚེ་གང་བསྐྱལ་ནའང་། རང་ཉིད་ཀྱི་ནང་དུ་འགྱུར་བ་རྩ་བའི་ཆ་ནས་གང་ཡང་མྱོང་མི་སྲིད།

ནང་ནས་མི་ཞིག་ལ་འགྱུར་བ་ཐེབས་པའི་ཤེས་བྱའི་རིགས་ནི་ཕྱིར་ལྟ་ཞིབ་བྱེད་པ་བརྒྱུད་ནས་ནམ་ཡང་སྒྲུབ་མི་ཐུབ།

ང་ཚོའི་ནང་དུ་རྩ་བའི་ཆ་ནས་ནང་གི་འགྱུར་བ་དངོས་སུ་བསྐྱེད་ཐུབ་པའི་ཤེས་བྱ་ངོ་མ་དེ་ལ་རང་ཉིད་ལ་ཐད་ཀར་རང་གིས་ལྟ་ཞིབ་བྱེད་པ་ནི་རྨང་གཞིར་འཛིན་དགོས།

ང་ཚོས་གསང་སྔགས་སློབ་ཕྲུག་ཚོར་རང་ཉིད་ལ་བལྟ་ཞིབ་བྱེད་དགོས་པ་དང་། ཁྱེད་ཚོས་རང་ལ་ལྟ་ཞིབ་ཇི་ལྟར་བྱེད་དགོས་པའི་དོན་དེ་དང་དེའི་རྒྱུ་མཚན་གང་ཡིན་ཞེས་བརྗོད་རྒྱུ་ནི་དགོས་ངེས་ཤིག་ཡིན།

ལྟ་ཞིབ་ནི་འཇིག་རྟེན་གྱི་འཕྲུལ་ཆས་ཀྱི་གནས་སྟངས་ལ་བསྒྱུར་བཅོས་བྱེད་པའི་ཐབས་ཤིག་ཡིན། ནང་གི་རང་ལ་ལྟ་ཞིབ་བྱེད་པ་ནི་ནང་ནས་འགྱུར་བ་གཏོང་བའི་ཐབས་ཤིག་ཡིན།

འདི་དག་ཐམས་ཅད་ཀྱི་རྗེས་འབྲས་སམ་ཡན་ལག་ལྟ་བུར་ང་ཚོས་ནན་བརྗོད་བྱེད་དགོས་པ་ནི་ཤེས་བྱའི་རིགས་གཉིས་ཡོད་དེ། ཕྱིའི་དང་ནང་གི་གཉིས་ཡོད་པ་དང་། གལ་ཏེ་ང་ཚོའི་ནང་དུ་ཤེས་བྱའི་ཁྱད་ཆོས་དབྱེ་བ་འབྱེད་ཐུབ་པའི་ལྟེ་བའི་སྟོབས་ཤུགས་མེད་ཚེ། བསམ་བློའི་རིམ་པ་དང་གོ་རིམ་གཉིས་ཀྱི་འདྲེས་སྦྱོར་འདིས་ང་ཚོ་མགོ་འཁོར་བའི་ནང་དུ་ཁྲིད་སྲིད།

འོག་ཏུ་ཚན་རིག་རིང་ལུགས་ཀྱི་རྟགས་མཚན་ཅན་གྱི་གསང་བའི་ལྟ་གྲུབ་དམན་པ་ནི་བལྟ་ཞིབ་བྱེད་ཆོག་པའི་ས་ཁོངས་དང་འབྲེལ་བ་ཡོད་ལ། འོན་ཀྱང་འདོད་པ་ཡོད་མཁན་མང་པོས་ནང་གི་ཤེས་བྱར་ངོས་འཛིན་བྱེད་ཀྱི་ཡོད།

ང་ཚོ་ཕྱིའི་དང་ནང་གི་འཇིག་རྟེན་གཉིས་ཀྱི་མདུན་དུ་འཕྲད་པ་ཡིན། འདི་དག་གི་དང་པོ་དེ་ནི་ཕྱིའི་ཚོར་བའི་དབང་པོས་ཚོར་བ་ཡིན། གཉིས་པ་དེ་ནི་ནང་གི་རང་ལ་རང་གིས་ལྟ་ཞིབ་བྱེད་པའི་ཚོར་བས་ཚོར་ཐུབ་པ་ཙམ་ཡིན།

བསམ་པ་དང་། བསམ་ཚུལ་དང་། སེམས་འགུལ་དང་། རེ་འདུན་དང་། རེ་བ་དང་། ཡིད་ཆེས་ཉམས་པ་སོགས་ནི་ནང་གི་ཡིན་ལ། སྤྱིར་བཏང་གི་ཚོར་བ་དང་རྒྱུན་ལྡན་གྱི་ཚོར་བས་མཐོང་མི་ཐུབ་ལ། འོན་ཀྱང་ང་ཚོའི་ཆེད་ཞལ་ལག་གི་ཅོག་ཙེ་དང་སྡོད་ཁང་གི་མལ་ཁྲི་ལས་ཀྱང་དོན་དངོས་ཆེ།

ངེས་པར་དུ་ང་ཚོ་ཕྱིའི་འཇིག་རྟེན་ལས་ནང་གི་འཇིག་རྟེན་ནང་དུ་འཚོ་བ་མང་བ་ཡིན། འདི་ནི་མི་འགོག་ཐུབ་པ་དང་མི་བསྙོན་ཐུབ་པ་ཞིག་ཡིན།

ང་ཚོའི་ནང་གི་འཇིག་རྟེན་ནང་དུ་སྟེ་ང་ཚོའི་གསང་བའི་འཇིག་རྟེན་ནང་དུ་ང་ཚོས་དགའ་ཞེན་དང་། འདོད་པ་དང་། དོགས་པ་དང་། བྱིན་རླབས་དང་། སྨོད་པ་དང་། རེ་འདུན་དང་། སྡུག་བསྔལ་དང་། སྐྱིད་སྣང་དང་། ཡིད་ཆེས་ཉམས་པ་དང་། གསོལ་རས་སོགས་མྱོང་བ་ཡིན།

ནང་གི་འཇིག་རྟེན་དང་ཕྱིའི་འཇིག་རྟེན་གཉིས་ཀ་ཚོད་ལྟའི་ཐོག་ནས་བདེན་དཔང་བྱེད་ཐུབ་པ་ཞིག་ཡིན་པ་ནི་བསྙོན་དུ་མེད་པ་ཞིག་ཡིན། ཕྱིའི་འཇིག་རྟེན་ནི་བལྟ་ཞིབ་བྱེད་ཆོག་པ་ཞིག་ཡིན། ནང་གི་འཇིག་རྟེན་ནི་རང་ཉིད་ལ་དང་རང་ཉིད་ཀྱི་ནང་དུ་འདི་ག་རང་དུ་ད་ལྟ་རང་ལ་རང་གིས་ལྟ་ཞིབ་བྱེད་ཆོག་པ་ཞིག་ཡིན།

སུ་ཞིག་གིས་བདེན་པ་བདེན་རྐྱང་གིས་གོ་ལ་འདིའི་ནང་གི་འཇིག་རྟེན་ནམ་ཉི་མའི་ཁྱིམ་ལུགས་སམ་ང་ཚོ་འཚོ་བའི་དཔལ་ཡོན་གྱི་ནང་གི་འཇིག་རྟེན་ལ་རྒྱུས་མངའ་ཡོད་དགོས་ན་སྔོན་ལ་ཁོ་རང་གི་ནང་གི་འཇིག་རྟེན་སྟེ་ཁོ་རང་གི་ནང་གི་འཚོ་བ་སྒེར་བ་སྟེ་ཁོ་རང་གི་ནང་གི་འཇིག་རྟེན་ལ་རྒྱུས་མངའ་ཡོད་དགོས།
