ང་ཚོའི་ཁྱིམ་དང་སློབ་གྲྭ་གཉིས་ཀར། ཕ་མ་དང་དགེ་རྒན་ཚོས་ང་ཚོར་ག་རེ་བསམ་དགོས་མིན་ནི་རྒྱུན་དུ་བཤད་ཀྱི་ཡོད་རུང་། ནམ་ཡང་ང་ཚོར་ཇི་ལྟར་བསམ་བློ་གཏོང་དགོས་མིན་སློབ་ཀྱི་མ་རེད།

ག་རེ་བསམ་དགོས་མིན་ཤེས་པ་ནི་ཅུང་ཙམ་སླ་པོ་ཡོད། ང་ཚོའི་ཕ་མ་དང་། དགེ་རྒན། སློབ་སྟོན་པ། དཔེ་དེབ་རྩོམ་པ་པོ་སོགས་ཚང་མ་རང་རང་གི་ལམ་ནས་བཙན་དབང་སྲིད་འཛིན་པ་རེ་རེད། ཚང་མས་ང་ཚོས་ཁོ་ཚོའི་བཙན་བཀའ་དང་། བླང་དོར། གཞུང་ལུགས། སྔོན་འཛིན་ལྟ་ཚུལ་སོགས་ལ་བསམ་བློ་གཏོང་འདོད་ཡོད།

བསམ་པའི་བཙན་དབང་སྲིད་འཛིན་པ་ནི་རྩྭ་ངན་བཞིན་དུ་མང་པོ་ཡོད། གཞན་གྱི་བསམ་པ་བྲན་དུ་བཀོལ་བ་དང་། བསམ་པ་བཅུམ་པ། བསམ་པ་ངེས་ཅན་གྱི་སྒྲིག་གཞི་དང་། སྔོན་འཛིན་ལྟ་ཚུལ། སློབ་གྲྭ་སོགས་ཀྱི་ནང་དུ་འཚོ་བཅུག་འདོད་པའི་འཁྱོག་བཤད་ཀྱི་འདུན་པ་ཞིག་གང་སར་ཡོད།

བསམ་པའི་བཙན་དབང་སྲིད་འཛིན་པ་ས་ཡ་དང་བྱེ་བ་མང་པོ་ཞིག་གིས་ནམ་ཡང་སུ་གང་གི་བསམ་པའི་རང་དབང་ལ་གུས་བརྩི་བྱེད་འདོད་མེད། གལ་སྲིད་སུ་ཞིག་ཁོ་ཚོ་དང་འདྲ་བའི་བསམ་བློ་མ་བཏང་ན། དེ་ནི་ངན་པ། ཕྱིར་ལོག་པ། མི་ཤེས་པ་སོགས་སུ་ངོས་འཛིན་བྱེད་ཀྱི་རེད།

ཚང་མས་ཚང་མ་བྲན་དུ་བཀོལ་འདོད་ཡོད་ལ། ཚང་མས་གཞན་དག་གི་ཤེས་རིག་གི་རང་དབང་ལ་རྡོག་རྫིས་གཏོང་འདོད་ཡོད། སུ་གཅིག་གིས་ཀྱང་གཞན་གྱི་བསམ་པའི་རང་དབང་ལ་གུས་བརྩི་བྱེད་འདོད་མེད། ཚང་མས་རང་ཉིད་བློ་དང་ལྡན་པ། མཁས་པ། ངོ་མཚར་ཅན་ཞིག་ཏུ་ཚོར་གྱི་ཡོད་ལ། ངང་གིས་གཞན་དག་རང་དང་འདྲ་བོ་ཡོང་བ་དང་། རང་ཉིད་དཔེ་བཞག་སྟེ་རང་དང་འདྲ་བའི་བསམ་བློ་གཏོང་དགོས་པར་འདོད་ཀྱི་རེད།

བསམ་པ་ལ་དབང་སྒྱུར་མང་དྲགས་བྱས་ཡོད། ཚོང་པ་དང་། ཚོང་ལས་ཁྱབ་བསྒྲགས་དུས་ཚགས་དང་། རླུང་འཕྲིན། བརྙན་འཕྲིན་སོགས་ནས་བྱེད་སྟངས་ལ་བལྟོས་དང་། ཚོང་ལས་ཁྱབ་བསྒྲགས་ནི་བཙན་དབང་སྲིད་འཛིན་གྱི་རྣམ་པས་བྱེད་ཀྱི་ཡོད། ཁྱེད་ཀྱིས་འདི་དང་འདི་འདྲ་བའི་རྡོ་སྣུམ་ཉོ། འདི་དང་འདི་འདྲ་བའི་ལྷམ་ཉོ། འདི་ཙམ་གྱི་རིན་པ་རེད། ཨ་སྒོར་འདི་ཙམ་རེད། ད་ལྟ་ཉོ། མྱུར་མྱུར་བྱས་ནས་ཉོ། སང་ཉིན་བར་དུ་འཇོག་མ་བྱེད། མྱུར་དུ་ཉོ་དགོས། མ་གཏོགས་ཁྱེད་ཀྱིས་བཀའ་མ་བརྩིས་ན་ང་ཚོས་ཁྱེད་བཙོན་ཁང་དུ་འཇུག་གི་ཡིན་ཟེར་བ་དང་། ཡང་ན་ཁྱེད་ལ་མི་གསོད་གཏོང་གི་ཡིན་ཟེར་བ་ཁོ་ན་མ་གཏོགས།

ཕ་མས་བུ་ཕྲུག་ལ་བསམ་བློ་ཤུགས་ཀྱིས་བཙན་འཇུག་བྱེད་འདོད་ཡོད་ལ། སློབ་གྲྭའི་དགེ་རྒན་གྱིས་བུ་དང་བུ་མོ་གཉིས་ཀས་དགེ་རྒན་གྱི་བསམ་བློ་བཙན་དབང་སྲིད་འཛིན་གྱི་རྣམ་པས་མ་བླངས་ཚེ། ཁོང་ཁྲོ་ལངས་པ་དང་། ཉེས་ཆད་གཅོད་པ། སྐར་གྲངས་དམའ་པོ་སྤྲོད་ཀྱི་རེད།

མི་ཕྱེད་ཀས་མི་ཕྱེད་ཀའི་བསམ་པ་བྲན་དུ་བཀོལ་འདོད་ཡོད། གཞན་དག་གི་བསམ་པ་བྲན་དུ་བཀོལ་འདོད་པའི་འདུན་པ་དེ་ང་ཚོས་ལོ་རྒྱུས་ནག་པོའི་ཤོག་ལྷེ་ལ་སློབ་སྦྱོང་བྱེད་དུས་མིག་ལ་མངོན་གསལ་དོད་པོ་མཐོང་ཐུབ།

གང་སར་མི་དམངས་བྲན་དུ་བཀོལ་རྒྱུར་བརྩོན་པའི་ཁྲག་སིམ་སིམ་གྱི་བཙན་དབང་སྲིད་འཛིན་པ་ཡོད་ལ་ད་ལྟའང་ཡོད། མི་དམངས་ཀྱིས་ག་རེ་བསམ་དགོས་མིན་བཙན་འགེལ་བྱེད་པའི་ཁྲག་སིམ་སིམ་གྱི་བཙན་དབང་སྲིད་འཛིན་པ། བསམ་བློ་རང་དབང་གཏོང་འདོད་བྱེད་མཁན་དེའི་སྐྱིད་སྡུག་ལ་བལྟོས་དང། དེ་ནི་མི་འགྱངས་པར་བཙོན་སྒར་དང་། སི་པི་རི་ཡ། བཙོན་ཁང་། ལས་ཀ་བཙན་བཀོལ། གཤོག་གཅོད། མེ་མདའ་རྒྱག་པ། ཕྱིར་འབུད་གཏོང་བ་སོགས་ལ་འགྲོ་གི་རེད།

དགེ་རྒན་དང་དགེ་རྒན་མ། ཕ་མ་ཚང་མས་ཇི་ལྟར་བསམ་བློ་གཏོང་དགོས་མིན་སློབ་འདོད་མེད།

མི་རྣམས་ཀྱིས་གཞན་ལ་རང་ཉིད་ཀྱིས་ཇི་ལྟར་དགོས་བསམ་པ་ལྟར་བསམ་བློ་གཏོང་དགོས་པའི་བཙན་བཀའ་གཏོང་རྒྱུར་དགའ་ཞིང་། དེའི་ནང་ཚང་མ་རང་རང་གི་ལམ་ནས་བཙན་དབང་སྲིད་འཛིན་པ་རེ་ཡིན་པ་གསལ་པོ་རེད། ཚང་མས་རང་ཉིད་མཐའ་མའི་ཐག་གཅོད་བྱེད་མཁན་དུ་བསམ་གྱི་ཡོད་ལ། ཚང་མས་གཞན་དག་ཚང་མས་རང་དང་འདྲ་བའི་བསམ་བློ་གཏོང་དགོས་པར་ཡིད་ཆེས་བརྟན་པོ་བྱེད་ཀྱི་རེད། གང་ཡིན་ཟེར་ན། ཁོ་ནི་ཡག་ཤོས་ནང་གི་ཡག་ཤོས་ཡིན་པས་རེད།
