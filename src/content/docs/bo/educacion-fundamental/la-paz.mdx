ཞི་བདེ་ནི་ལས་འཆར་དང་རྒྱལ་སྤྱིའི་ཉེན་རྟོག་པ། མཉམ་འབྲེལ་རྒྱལ་ཚོགས། ཨོ་ཨེ་ཨེ། རྒྱལ་སྤྱིའི་ཆིངས་ཡིག ཡང་ན་ཞི་བདེའི་མིང་ཁྱེར་ནས་བཙན་འཛུལ་བྱེད་མཁན་གྱི་དམག་དཔུང་དང་འབྲེལ་བ་མེད།

གལ་སྲིད་ང་ཚོས་བདེན་པའི་ཞི་བདེ་འདོད་ན་དམག་འཁྲུག་གི་སྐབས་སུ་ལྟ་སྲུང་བྱེད་མཁན་ཇི་བཞིན་དུ་གནས་དགོས་ཏེ། དུས་རྟག་ཏུ་དོགས་ཟོན་དང་ལྟ་སྲུང་བྱེད་པ། བསམ་པ་གྲུང་སྟོབས་ལྡན་པ། རྒྱུ་མཚན་ནི་ཞི་བདེ་ནི་རྨི་ལམ་སྙན་མོ་དང་རོལ་དབྱངས་ཀྱི་འཁྲུལ་སྣང་མ་ཡིན་པའི་རྐྱེན་གྱིས་རེད།

གལ་སྲིད་ང་ཚོས་སྐད་ཅིག་སྐད་ཅིག་ལ་དོགས་ཟོན་བྱེད་པའི་གནས་སྟངས་སུ་གནས་སྟངས་སུ་འཚོ་བ་བསྐྱེལ་སྟངས་མ་ཤེས་ན། ཞི་བདེའི་ཕྱོགས་སུ་འཁྲིད་པའི་ལམ་དེ་ནི་སླ་བོ་ཞིག་མིན་པར་དོག་པོར་འགྱུར། དེ་ནས་ཧ་ཅང་ཁག་པོར་ཆགས་རྗེས་མཐའ་མཇུག་ཏུ་འགྲོ་ས་མེད་པའི་ལམ་ཁ་ཞིག་ཏུ་ཐོན་ངེས་རེད།

ང་ཚོས་ངེས་པར་དུ་ཧ་གོ་དགོས་པ་ནི་སེམས་དྭངས་གཙང་གི་བདེན་པའི་ཞི་བདེ་དེ་ནི་ང་ཚོ་སླེབས་ཐུབ་པའི་ཁྱིམ་ཚང་ཞིག་དང་བུ་མོ་མཛེས་མ་ཞིག་གིས་དགའ་སྤྲོས་ངང་སྒུག་སྲིད་པ་ཞིག་མིན། ཞི་བདེ་ནི་དམིགས་འབེན་དང་ས་ཆ། ཡང་ན་དེ་དང་འབྲེལ་བ་ཡོད་པའི་གནས་སྟངས་ཤིག་མིན།

ཞི་བདེའི་རྗེས་སུ་འབྲང་བ་དང་འཚོལ་བ། དེའི་ཐོག་ཏུ་ལས་འཆར་བཟོ་བ། དེའི་མིང་ཁྱེར་ནས་འཐབ་རྩོད་བྱེད་པ། དེའི་སྐོར་ལ་དྲིལ་བསྒྲགས་བྱེད་པ། དེའི་ཆེད་དུ་ལས་ཁུངས་གསར་འཛུགས་བྱེད་པ་སོགས་ནི་རྦད་དེ་དོན་མེད་རེད། རྒྱུ་མཚན་ནི་ཞི་བདེ་ནི་བསམ་པའི་ཆ་ཤས་མིན་པའི་རྐྱེན་གྱིས་རེད། ཞི་བདེ་ནི་སེམས་དྭངས་གཙང་གི་དྲི་ཞིམ་འཐུལ་བའི་མེ་ཏོག་དང་མཚུངས་སོ། །

ཞི་བདེ་ནི་ཉོ་ཚོང་བྱེད་མི་ཐུབ་ལ་ཞི་འཇགས་སུ་གཏོང་བའི་ལམ་ལུགས་དང་དམིགས་བསལ་ཚོད་འཛིན། ཉེན་རྟོག་པ་སོགས་ཀྱི་སྒོ་ནས་ཀྱང་ཐོབ་མི་ཐུབ།

རྒྱལ་ཁབ་ཁ་ཤས་ནང་དུ་རྒྱལ་ཡོངས་དམག་དཔུང་གིས་ཞིང་ཁ་རྣམས་སྟོར་བརླག་བཏང་ནས་མི་རྣམས་བསད་པ་དང་ཇག་རྐུན་ཡིན་པའི་དོགས་གཞི་ཡོད་མཁན་རྣམས་མེ་མདའ་བརྒྱབ་ནས་བསད་པ་རྣམས་ཞི་བདེའི་མིང་ཁྱེར་ནས་བྱེད་ཀྱི་ཡོད། བྱ་སྤྱོད་དེ་འདྲ་ཞིག་གི་མཇུག་འབྲས་ནི་གདུག་རྩུབ་ཀྱི་རང་བཞིན་ཇེ་མང་དུ་འགྲོ་རྒྱུ་དེ་རེད།

དྲག་ཤུགས་ལས་དྲག་ཤུགས་སྐྱེ་བ་དང་སྡང་ཟུག་ལས་སྡང་ཟུག་སྐྱེ་ཡི་རེད། ཞི་བདེ་ནི་བཙན་འཛུལ་བྱས་ནས་ཐོབ་མི་ཐུབ་ལ། ཞི་བདེ་ནི་དྲག་ཤུགས་ཀྱི་མཇུག་འབྲས་ཀྱང་ཡིན་མི་སྲིད། ཞི་བདེ་ནི་ང་ཚོས་རང་ཉིད་ཀྱི་"ང་"ཞེས་པ་དེ་མེད་པར་བཟོས་དུས་ད་གཟོད་ང་ཚོའི་སར་སླེབས། ང་ཚོས་རང་ཉིད་ཀྱི་ནང་དུ་དམག་འཁྲུག་བཟོ་བའི་སེམས་ཁམས་ཀྱི་རྒྱུ་རྐྱེན་དེ་དག་མེད་པར་བཟོས་དུས་ང་ཚོའི་སར་སླེབས་ཀྱི་རེད།

གལ་སྲིད་ང་ཚོས་ཞི་བདེ་འདོད་ན་བརྟག་དཔྱད་དང་སློབ་སྦྱོང་བྱེད་དགོས། དེ་བཞིན་དུ་བརྟག་དཔྱད་དང་སློབ་སྦྱོང་བྱེད་དགོས། ང་ཚོས་བརྟག་དཔྱད་དང་སློབ་སྦྱོང་བྱས་ནས་རི་མོའི་ཟུར་ཁྱོགས་གཅིག་མ་ཡིན་པར་སྤྱིའི་རྣམ་པ་དེ་མཐོང་ཐུབ་དགོས།

ང་ཚོའི་ནང་དུ་ནང་གི་རྣམ་པ་རྦད་དེ་བསྒྱུར་བ་བཏང་དུས་ཞི་བདེ་སྐྱེ་ཡི་རེད།

ཚོད་འཛིན་དང་ཞི་བདེའི་ལས་ཁུངས། ཞི་འཇགས་སུ་གཏོང་བ་སོགས་ཀྱི་གནད་དོན་དེ་དག་ནི་འཚོ་བའི་རྒྱ་མཚོའི་ནང་གི་ཐ་དད་པའི་ཞིབ་ཕྲ་རེ་རེ་དང་གནས་སྟངས་རེ་རེ། དེ་བཞིན་དུ་མི་ཚེའི་སྤྱིའི་རྣམ་པའི་ཐ་དད་པའི་ཆ་ཤས་ཤིག་ཡིན་སྟབས་དེ་དག་གིས་ནམ་ཡང་ཞི་བདེའི་གནད་དོན་དེ་རྩ་བའི་ཆ་ནས་དང་སྤྱི་ཡོངས་དང་མཐའ་མའི་ཆ་ནས་སེལ་ཐུབ་ཀྱི་མ་རེད།

ང་ཚོས་རི་མོ་དེ་ཆ་ཚང་བའི་རྣམ་པའི་སྒོ་ནས་བལྟ་དགོས། འཇིག་རྟེན་གྱི་གནད་དོན་ནི་མི་སྒེར་གྱི་གནད་དོན་རེད། གལ་སྲིད་མི་སྒེར་ལ་ནང་སེམས་སུ་ཞི་བདེ་མེད་ན་སྤྱི་ཚོགས་དང་འཇིག་རྟེན་དེ་མི་འགྱུར་བའི་དམག་འཁྲུག་གི་ནང་དུ་གནས་ངེས་རེད།
