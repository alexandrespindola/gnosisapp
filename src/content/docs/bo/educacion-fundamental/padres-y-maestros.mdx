གལ་ཏེ་ཕ་མ་དང་དགེ་རྒན་རང་ཉིད་མི་ཤེས་ན། བྱིས་པ་དང་བྱིས་མོ་ཚོར་གོ་བ་ལེན་མི་ཐུབ་ན། འཚོ་བ་གསར་པ་འགོ་འཛུགས་བྱེད་མཁན་གྱི་སྲོག་ཆགས་ཆུང་ཆུང་འདི་ཚོ་དང་འབྲེལ་བ་གཏིང་ཟབ་ཇི་ལྟར་ཡོད་མེད་ཤེས་མི་ཐུབ་ན། ཁོ་ཚོའི་སློབ་མ་ཚོའི་བློ་གྲོས་འདོན་སྤེལ་བྱེད་རྒྱུའི་ཐད་ལ་སེམས་ཁྲལ་བྱེད་པ་ཙམ་བྱས་ན། ང་ཚོས་ཤེས་ཡོན་གྱི་རིམ་པ་གསར་པ་ཞིག་ཇི་ལྟར་བསྐྲུན་ཐུབ་བམ།

བྱིས་པ་དང་སློབ་མ་ཚོ་སློབ་གྲྭར་འདུ་ཤེས་དང་ལྡན་པའི་ལམ་སྟོན་ལེན་པར་འགྲོ་ཡི་རེད། འོན་ཀྱང་གལ་ཏེ་དགེ་རྒན་ཚོ་བསམ་ཚུལ་དོག་ཅིང་རྙིང་ཞེན་ཅན། འགྱུར་བ་མི་འདོད་མཁན་ཞིག་ཡིན་ཚེ། སློབ་མ་ཡང་དེ་འདྲ་ཞིག་ཏུ་འགྱུར་སྲིད།

ཤེས་ཡོན་སློབ་གསོ་སྤྲོད་མཁན་ཚོས་སློབ་སྦྱོང་བསྐྱར་ཟློས་དང་རང་ཉིད་ཤེས་པར་བྱེད་དགོས། ཁོ་ཚོས་ཤེས་ཡོན་ཡོད་ཚད་ཞིབ་བཤེར་བྱེད་དགོས། ང་ཚོ་དུས་རབས་གསར་པ་ཞིག་གི་ནང་དུ་སླེབས་ཀྱི་ཡོད་པ་ཧ་གོ་དགོས།

ཤེས་ཡོན་སློབ་གསོ་སྤྲོད་མཁན་ཚོ་འགྱུར་བ་ཕྱིན་ན་སྤྱི་པའི་ཤེས་ཡོན་ཡང་འགྱུར་བ་འགྲོ་ཡི་རེད།

ཤེས་ཡོན་སློབ་གསོ་སྤྲོད་མཁན་ལ་སློབ་གསོ་སྤྲོད་རྒྱུ་དེ་ཁག་ཤོས་རེད། གང་ལྟར་ཀློག་རྒྱུ་མང་པོ་ཡོད་མཁན་དང་། ཕྱག་འཁྱེར་ཡོད་མཁན། སློབ་ཁྲིད་བྱེད་དགོས་མཁན་སློབ་གྲྭའི་དགེ་རྒན་ལས་ཀ་བྱེད་མཁན་ཚང་མ་ད་ག་རང་རེད། ཁོ་ཚོའི་བསམ་བློ་དེ་ཁོ་ཚོས་སློབ་སྦྱོང་བྱས་པའི་བསམ་བཞིན་ཁྲི་ལྔ་བཅུའི་ནང་དུ་བཅུག་ནས་ཡོད་པ་རེད། ད་ཁོ་ཚོ་མེ་མདའི་སྒྲ་གྲགས་ནས་ཀྱང་འགྱུར་བ་འགྲོ་གི་མ་རེད།

དགེ་རྒན་རྣམས་ཀྱིས་ཇི་ལྟར་བསམ་བློ་གཏོང་དགོས་མིན་བསླབ་དགོས་ཀྱི་ཡོད་རེད། ཡིན་ནའང་ཁོ་ཚོས་སེམས་ཁྲལ་བྱེད་ས་དེ་ཁོ་ཚོས་གང་ཞིག་ལ་བསམ་བློ་གཏོང་དགོས་མིན་དེ་རེད།

ཕ་མ་དང་དགེ་རྒན་ཚོ་དཔལ་འབྱོར་དང་སྤྱི་ཚོགས། སེམས་ཚོར་དང་འབྲེལ་བའི་སེམས་ཁྲལ་འཇིགས་སུ་རུང་བ་མང་པོས་ཁེངས་ནས་ཡོད།

ཕ་མ་དང་དགེ་རྒན་ཚོ་ཕལ་ཆེར་རང་ཉིད་ཀྱི་དཀའ་ངལ་དང་སྡུག་བསྔལ་གྱིས་ཁེངས་ནས་ཡོད། ཁོ་ཚོས་དོན་དངོས་སུ་དོ་སྣང་བྱེད་ས་དེ་གཞོན་ནུ་དང་གཞོན་ནུ་མའི་ "རླབས་གསར་པའི་" དཀའ་ངལ་ལ་སློབ་སྦྱོང་བྱེད་པ་དང་ཐག་གཅོད་རྒྱུར་མིན་ནོ།

བསམ་བློ་དང་བཟང་སྤྱོད། སྤྱི་ཚོགས་ཀྱི་ཉམས་རྒུད་འཇིགས་སུ་རུང་བ་ཞིག་ཡོད་པ་རེད། འོན་ཀྱང་ཕ་མ་དང་དགེ་རྒན་ཚོ་སྒེར་གྱི་སེམས་འཁྲལ་དང་སེམས་ཁྲལ་གྱིས་ཁེངས་ནས་ཡོད་ལ། དུས་ཚོད་ཡོད་ཚད་བྱིས་པའི་དཔལ་འབྱོར་གྱི་ཕྱོགས་ལ་བསམ་བློ་གཏོང་བ་དང་། མུ་གེའི་ངང་འཆི་མི་དགོས་པའི་ཆེད་ལས་ཤིག་སྤྲོད་རྒྱུ་དེ་ཙམ་རེད།

ཡོངས་གྲགས་ཀྱི་ཡིད་ཆེས་དང་ལྡོག་སྟེ། ཁྱིམ་ཚང་གི་ཕ་མའི་མང་ཆེ་བས་བྱིས་པར་དངོས་འབྲེལ་གྱི་བྱམས་བརྩེ་མི་བྱེད། གལ་ཏེ་བྱམས་བརྩེ་བྱེད་ན། སྤྱི་པའི་བདེ་རྩའི་ཆེད་དུ་འཐབ་རྩོད་བྱེད་ཀྱི་རེད། སྤྱི་པའི་ཤེས་ཡོན་གྱི་དཀའ་ངལ་ལ་དོ་སྣང་བྱེད་ཀྱི་རེད། དམིགས་ཡུལ་དེ་འགྱུར་བ་ངོ་མ་ཞིག་སྒྲུབ་རྒྱུ་རེད།

གལ་ཏེ་ཁྱིམ་ཚང་གི་ཕ་མ་ཚོས་བྱིས་པར་དངོས་འབྲེལ་གྱི་བྱམས་བརྩེ་བྱེད་ན། དམག་འཁྲུག་ཡོང་གི་མ་རེད། ཁྱིམ་ཚང་དང་རྒྱལ་ཁབ་ལ་འཛམ་གླིང་ཡོངས་སུ་ཁྱབ་པའི་ངོ་རྒོལ་དེ་ཙམ་གྱིས་གཙིགས་ཆེན་བྱེད་ཀྱི་མ་རེད། གང་ཡིན་ཟེར་ན། འདིས་དཀའ་ངལ་དང་དམག་འཁྲུགས། གནོད་འཚེ་ཅན་གྱི་དབྱེ་འབྱེད། ང་ཚོའི་བུ་དང་བུ་མོ་ཚོའི་ཆེད་དུ་དམྱལ་བའི་ཁོར་ཡུག་བསྐྲུན་གྱི་རེད།

མི་ཚོས་སྨན་པ་དང་བཟོ་བསྐྲུན་པ། ཁྲིམས་རྩོད་པ་སོགས་བྱེད་པར་སློབ་སྦྱོང་དང་གྲ་སྒྲིག་བྱེད་ཀྱི་རེད། དེ་ལས་ལྡོག་སྟེ་ཁྱིམ་ཚང་གི་ཕ་མ་བྱེད་པའི་ལས་དོན་དེ་དཀའ་ཤོས་དང་གལ་ཆེ་ཤོས་ཡིན་པའི་གྲ་སྒྲིག་བྱེད་ཀྱི་མེད།

ཁྱིམ་ཚང་གི་བདག་འཛིན་དང་ང་ཚོའི་ཉེ་འབྲེལ་ལ་བྱམས་བརྩེ་མེད་པ། ཁྱིམ་ཚང་གི་ཁེར་རྐྱང་གི་སྲིད་བྱུས་དེ་བརྒྱ་ཆ་བརྒྱ་ཐམ་པའི་ཐོག་ནས་དོན་མེད་རེད། གང་ཡིན་ཟེར་ན། དེ་ཉམས་རྒུད་དུ་གཏོང་མཁན་དང་སྤྱི་ཚོགས་ཡར་རྒྱས་འགྲོ་བར་འགོག་རྐྱེན་བྱེད་མཁན་ཞིག་ཏུ་འགྱུར་གྱི་རེད།
