ຈິດຕະວິທະຍາການປະຕິວັດຍຸກໃໝ່ຢືນຢັນວ່າເຄື່ອງຈັກອົງຄະທາດຂອງສັດທີ່ມີຄວາມຄິດທີ່ເອີ້ນວ່າຄົນຢ່າງບໍ່ຖືກຕ້ອງ, ມີຢູ່ໃນຮູບແບບສາມສູນ ຫຼື ສາມສະໝອງ.

ສະໝອງທໍາອິດຖືກປິດຢູ່ໃນກະໂຫຼກຫົວ. ສະໝອງທີສອງກົງກັບກະດູກສັນຫຼັງໂດຍສະເພາະກັບໄຂກະດູກສັນຫຼັງກາງແລະສາຂາເສັ້ນປະສາດທັງໝົດ. ສະໝອງທີສາມບໍ່ໄດ້ຢູ່ໃນສະຖານທີ່ກໍານົດຫຼືແມ່ນແຕ່ອະໄວຍະວະໃດນຶ່ງ. ຕົວຈິງແລ້ວສະໝອງທີສາມປະກອບດ້ວຍເສັ້ນປະສາດ sympathetic ແລະໂດຍທົ່ວໄປທັງໝົດສູນເສັ້ນປະສາດສະເພາະຂອງຮ່າງກາຍມະນຸດ.

ສະໝອງທໍາອິດແມ່ນສູນກາງແຫ່ງຄວາມຄິດ. ສະໝອງທີສອງແມ່ນສູນກາງຂອງການເຄື່ອນໄຫວ, ໂດຍທົ່ວໄປເອີ້ນວ່າສູນກາງມໍເຕີ. ສະໝອງທີສາມແມ່ນສູນກາງຂອງອາລົມ.

ໄດ້ຮັບການພິສູດຢ່າງສົມບູນໃນການປະຕິບັດວ່າການໃຊ້ສະໝອງຄິດຫຼາຍເກີນໄປເຮັດໃຫ້ເກີດການໃຊ້ພະລັງງານທາງປັນຍາຫຼາຍເກີນໄປ. ດັ່ງນັ້ນ, ມັນສົມເຫດສົມຜົນທີ່ຈະເວົ້າໂດຍບໍ່ຕ້ອງຢ້ານວ່າໂຮງໝໍບ້າແມ່ນສຸສານທີ່ແທ້ຈິງຂອງຄົນຕາຍທາງປັນຍາ.

ກິລາທີ່ກົມກຽວແລະສົມດູນແມ່ນເປັນປະໂຫຍດສໍາລັບສະໝອງມໍເຕີ, ແຕ່ການໃຊ້ກິລາຫຼາຍເກີນໄປໝາຍເຖິງການໃຊ້ພະລັງງານມໍເຕີຫຼາຍເກີນໄປແລະຜົນໄດ້ຮັບມັກຈະຮ້າຍແຮງ. ມັນບໍ່ແມ່ນເລື່ອງທີ່ບໍ່ມີເຫດຜົນທີ່ຈະເວົ້າວ່າມີຄົນຕາຍຂອງສະໝອງມໍເຕີ. ຄົນຕາຍເຫຼົ່ານັ້ນເປັນທີ່ຮູ້ຈັກກັນໃນນາມຄົນເຈັບທີ່ເປັນພະຍາດ Hemiplegia, Paraplegia, Parálisis progressive, ແລະອື່ນໆ.

ຄວາມຮູ້ສຶກດ້ານຄວາມງາມ, ຄວາມລຶກລັບ, ຄວາມສຸກສູງສຸດ, ດົນຕີທີ່ສູງກວ່າແມ່ນຈໍາເປັນເພື່ອປູກຝັງສູນກາງອາລົມ, ແຕ່ການໃຊ້ສະໝອງນັ້ນຫຼາຍເກີນໄປເຮັດໃຫ້ເກີດການສວມໃສ່ທີ່ບໍ່ຈໍາເປັນແລະການສູນເສຍພະລັງງານທາງດ້ານອາລົມ. ພວກ existentialists ຂອງ "ຄື້ນໃໝ່", ຜູ້ທີ່ມັກ Rock, ນັກສິລະປິນ sensual ປອມຂອງສິລະປະທີ່ທັນສະໄໝ, ຜູ້ທີ່ມັກ morbid ຂອງຄວາມຮູ້ສຶກ, ແລະອື່ນໆ, ແລະອື່ນໆ, ໃຊ້ສະໝອງທາງດ້ານອາລົມໃນທາງທີ່ຜິດ.

ເຖິງແມ່ນວ່າເບິ່ງຄືວ່າບໍ່ໜ້າເຊື່ອ, ການຕາຍແນ່ນອນເກີດຂຶ້ນໂດຍສ່ວນສາມໃນແຕ່ລະຄົນ. ໄດ້ຮັບການພິສູດຢ່າງພຽງພໍວ່າທຸກໆພະຍາດມີພື້ນຖານຂອງມັນຢູ່ໃນສະໝອງທັງສາມຢ່າງໃດນຶ່ງ.

ກົດໝາຍທີ່ຍິ່ງໃຫຍ່ໄດ້ຝາກໄວ້ຢ່າງສະຫຼາດສະຫຼຽວໃນແຕ່ລະສະໝອງທັງສາມຂອງສັດທີ່ມີຄວາມຄິດ, ທຶນທີ່ກໍານົດໄວ້ຂອງຄຸນຄ່າທີ່ສໍາຄັນ. ການປະຢັດທຶນນັ້ນໝາຍເຖິງການຍືດອາຍຸ, ການສູນເສຍທຶນນັ້ນເຮັດໃຫ້ເກີດການເສຍຊີວິດ.

ປະເພນີເກົ່າແກ່ທີ່ມາຮອດພວກເຮົາຈາກຄືນທີ່ໜ້າຢ້ານຂອງສັດຕະວັດແລ້ວ, ຢືນຢັນວ່າອາຍຸສະເລ່ຍຂອງມະນຸດໃນທະວີບ MU ເກົ່າ, ຕັ້ງຢູ່ໃນມະຫາສະໝຸດປາຊີຟິກ, ຕັ້ງແຕ່ສິບສອງຫາສິບຫ້າສັດຕະວັດ.

ເມື່ອຫຼາຍສັດຕະວັດຜ່ານມາຕະຫຼອດທຸກຍຸກທຸກສະໄໝ ການນໍາໃຊ້ສະໝອງທັງສາມຢ່າງທີ່ບໍ່ຖືກຕ້ອງໄດ້ເຮັດໃຫ້ຊີວິດສັ້ນລົງເທື່ອລະກ້າວ.

ຢູ່ໃນປະເທດທີ່ມີແສງແດດຂອງ KEM ... ຢູ່ທີ່ປະເທດເອຢິບເກົ່າແກ່ຂອງພວກຟາໂຣ ອາຍຸສະເລ່ຍຂອງມະນຸດບັນລຸພຽງແຕ່ຮ້ອຍສີ່ສິບປີເທົ່ານັ້ນ.

ປະຈຸບັນໃນຍຸກສະໄໝໃໝ່ເຫຼົ່ານີ້ຂອງນໍ້າມັນແອັດຊັງແລະ cellulose, ໃນຍຸກຂອງ existentialism ແລະ rebels of Rock, ອາຍຸສະເລ່ຍຂອງມະນຸດອີງຕາມບາງບໍລິສັດປະກັນໄພ, ແມ່ນພຽງແຕ່ຫ້າສິບປີເທົ່ານັ້ນ.

ພວກທ່ານ Marxists-Leninists ຂອງສະຫະພາບໂຊວຽດ, ເປັນຄົນຂີ້ຕົວະແລະຂີ້ຕົວະສະເໝີ, ບອກວ່າພວກເຂົາໄດ້ປະດິດ serums ພິເສດເພື່ອຍືດອາຍຸ, ແຕ່ເຖົ້າ Kruschev ຍັງບໍ່ທັນມີອາຍຸແປດສິບປີແລະຕ້ອງຂໍອະນຸຍາດຕີນເບື້ອງໜຶ່ງເພື່ອຍົກຕີນອີກເບື້ອງໜຶ່ງ.

ໃນໃຈກາງຂອງອາຊີມີຊຸມຊົນທາງສາສະໜາທີ່ປະກອບດ້ວຍຜູ້ເຖົ້າຜູ້ແກ່ທີ່ບໍ່ຈື່ຈໍາໄວໜຸ່ມຂອງພວກເຂົາ. ອາຍຸສະເລ່ຍຂອງຜູ້ເຖົ້າຜູ້ແກ່ເຫຼົ່ານັ້ນຕັ້ງແຕ່ສີ່ຮ້ອຍຫາຫ້າຮ້ອຍປີ.

ຄວາມລັບທັງໝົດຂອງອາຍຸຍືນຂອງພະສົງອາຊີເຫຼົ່ານີ້ແມ່ນຂຶ້ນກັບການນໍາໃຊ້ສະໝອງທັງສາມຢ່າງທີ່ສະຫຼາດສະຫຼຽວ.

ການເຮັດວຽກທີ່ສົມດູນແລະກົມກຽວຂອງສະໝອງທັງສາມຢ່າງໝາຍເຖິງການປະຢັດຄຸນຄ່າທີ່ສໍາຄັນແລະເປັນລໍາດັບຢ່າງມີເຫດຜົນ, ການຂະຫຍາຍຊີວິດ.

ມີກົດໝາຍ cosmic ທີ່ເອີ້ນວ່າ "ການເທົ່າທຽມກັນຂອງການສັ່ນສະເທືອນຂອງຫຼາຍແຫຼ່ງ". ພະສົງຂອງວັດແຫ່ງນັ້ນຮູ້ວິທີນໍາໃຊ້ກົດໝາຍນັ້ນໂດຍການໃຊ້ສະໝອງທັງສາມຢ່າງ.

ການສິດສອນທີ່ບໍ່ເໝາະສົມນໍາພານັກຮຽນຊາຍແລະຍິງໄປສູ່ການໃຊ້ສະໝອງຄິດຫຼາຍເກີນໄປ ເຊິ່ງຜົນໄດ້ຮັບທີ່ຈິດຕະສາດຮູ້ແລ້ວ.

ການປູກຝັງທີ່ສະຫຼາດສະຫຼຽວຂອງສະໝອງທັງສາມຢ່າງແມ່ນການສຶກສາພື້ນຖານ. ໃນໂຮງຮຽນເກົ່າແກ່ຂອງຄວາມລຶກລັບຂອງບາບີໂລນ, ປະເທດເກຣັກ, ອິນເດຍ, ເປີເຊຍ, ອີຢິບ, ແລະອື່ນໆ, ນັກຮຽນຊາຍແລະຍິງໄດ້ຮັບຂໍ້ມູນທີ່ສົມບູນແບບໂດຍກົງສໍາລັບສະໝອງທັງສາມຂອງພວກເຂົາໂດຍຜ່ານຄໍາສັ່ງ, ການເຕັ້ນ, ດົນຕີ, ແລະອື່ນໆ, ປະສົມປະສານຢ່າງສະຫຼາດສະຫຼຽວ.

ໂຮງລະຄອນຂອງສະໄໝບູຮານເປັນສ່ວນໜຶ່ງຂອງໂຮງຮຽນ. ລະຄອນ, ຕະຫຼົກ, ໂສກນາດຕະກໍາ, ລວມກັບການສະແດງ mime ພິເສດ, ດົນຕີ, ການສອນທາງປາກ, ແລະອື່ນໆ. ມັນຖືກນໍາໃຊ້ເພື່ອແຈ້ງໃຫ້ສະໝອງທັງສາມຂອງແຕ່ລະຄົນ.

ຫຼັງຈາກນັ້ນ, ນັກຮຽນບໍ່ໄດ້ໃຊ້ສະໝອງຄິດຫຼາຍເກີນໄປແລະຮູ້ວິທີການໃຊ້ສະໝອງທັງສາມຂອງພວກເຂົາຢ່າງສະຫຼາດສະຫຼຽວແລະສົມດູນ.

ການເຕັ້ນຂອງຄວາມລຶກລັບຂອງ Eleusis ໃນປະເທດເກຣັກ, ໂຮງລະຄອນໃນບາບີໂລນ, ຮູບປັ້ນໃນປະເທດເກຣັກໄດ້ຖືກນໍາໃຊ້ສະເໝີເພື່ອຖ່າຍທອດຄວາມຮູ້ໃຫ້ແກ່ຜູ້ຕິດຕາມຊາຍແລະຍິງ.

ຕອນນີ້ຢູ່ໃນຍຸກທີ່ເສື່ອມໂຊມຂອງ Rock, ນັກຮຽນຊາຍແລະຍິງສັບສົນແລະສູນເສຍທິດທາງຍ່າງໄປຕາມເສັ້ນທາງທີ່ມືດມົນຂອງການທາລຸນທາງຈິດໃຈ.

ປະຈຸບັນບໍ່ມີລະບົບຜູ້ສ້າງທີ່ແທ້ຈິງສໍາລັບການປູກຝັງກົມກຽວຂອງສະໝອງທັງສາມຢ່າງ.

ຄູສອນຊາຍແລະຍິງຂອງໂຮງຮຽນ, ວິທະຍາໄລແລະມະຫາວິທະຍາໄລ, ພຽງແຕ່ແນເປົ້າໃສ່ຄວາມຊົງຈໍາທີ່ບໍ່ຊື່ສັດຂອງນັກຮຽນທີ່ເບື່ອຫນ່າຍທີ່ລໍຖ້າດ້ວຍຄວາມກັງວົນສໍາລັບເວລາທີ່ຈະອອກຈາກຫ້ອງຮຽນ.

ມັນຮີບດ່ວນ, ມັນເປັນສິ່ງຈໍາເປັນທີ່ຈະຮູ້ວິທີການປະສົມປະສານປັນຍາ, ການເຄື່ອນໄຫວແລະອາລົມເພື່ອຈຸດປະສົງຂອງການນໍາເອົາຂໍ້ມູນທີ່ສົມບູນແບບໄປສູ່ສະໝອງທັງສາມຂອງນັກຮຽນ.

ມັນເປັນເລື່ອງທີ່ບໍ່ມີເຫດຜົນທີ່ຈະແຈ້ງໃຫ້ສະໝອງດຽວ. ສະໝອງທໍາອິດບໍ່ແມ່ນສະໝອງດຽວຂອງການຮັບຮູ້. ມັນເປັນເລື່ອງທີ່ເປັນອາດຊະຍາກໍາທີ່ຈະໃຊ້ສະໝອງຄິດຂອງນັກຮຽນຊາຍແລະຍິງໃນທາງທີ່ຜິດ.

ການສຶກສາພື້ນຖານຄວນນໍານັກຮຽນໄປຕາມເສັ້ນທາງແຫ່ງການພັດທະນາທີ່ກົມກຽວ.

ຈິດຕະວິທະຍາການປະຕິວັດສອນຢ່າງຊັດເຈນວ່າສະໝອງທັງສາມມີສາມປະເພດຂອງສະມາຄົມທີ່ເປັນເອກະລາດທີ່ແຕກຕ່າງກັນຢ່າງສິ້ນເຊີງ. ສາມປະເພດຂອງສະມາຄົມເຫຼົ່ານີ້ evoke ປະເພດທີ່ແຕກຕ່າງກັນຂອງແຮງກະຕຸ້ນຂອງການເປັນ.

ອັນນີ້ເຮັດໃຫ້ພວກເຮົາຢູ່ໃນຄວາມເປັນຈິງສາມບຸກຄະລິກທີ່ແຕກຕ່າງກັນທີ່ບໍ່ມີຫຍັງທົ່ວໄປທັງໃນລັກສະນະຂອງມັນຫຼືໃນການສະແດງອອກຂອງມັນ.

ຈິດຕະວິທະຍາການປະຕິວັດຍຸກໃໝ່ສອນວ່າໃນແຕ່ລະຄົນມີສາມດ້ານທາງດ້ານຈິດໃຈທີ່ແຕກຕ່າງກັນ. ດ້ວຍສ່ວນໜຶ່ງຂອງແກ່ນທາງດ້ານຈິດໃຈ, ພວກເຮົາຕ້ອງການສິ່ງໜຶ່ງ, ດ້ວຍສ່ວນອື່ນ, ພວກເຮົາຕ້ອງການສິ່ງທີ່ແຕກຕ່າງກັນຢ່າງຕັດສິນ ແລະ ຂອບໃຈສ່ວນທີສາມ, ພວກເຮົາເຮັດບາງສິ່ງບາງຢ່າງທີ່ກົງກັນຂ້າມຢ່າງສິ້ນເຊີງ.

ໃນຊ່ວງເວລາແຫ່ງຄວາມເຈັບປວດອັນສູງສຸດ, ບາງທີການສູນເສຍຄົນທີ່ຮັກຫຼືໄພພິບັດໃດນຶ່ງທີ່ໃກ້ຊິດ, ບຸກຄະລິກກະພາບທາງດ້ານອາລົມໄປສູ່ຄວາມສິ້ນຫວັງໃນຂະນະທີ່ບຸກຄະລິກກະພາບທາງປັນຍາຖາມວ່າເປັນຫຍັງໂສກນາດຕະກໍາທັງໝົດນັ້ນແລະບຸກຄະລິກກະພາບຂອງການເຄື່ອນໄຫວພຽງແຕ່ຕ້ອງການທີ່ຈະໜີຈາກສາກ.

ສາມບຸກຄະລິກທີ່ແຕກຕ່າງກັນ, ແຕກຕ່າງກັນແລະບາງຄັ້ງກໍ່ກົງກັນຂ້າມກັນຫຼາຍຄັ້ງ ຄວນໄດ້ຮັບການປູກຝັງແລະສອນຢ່າງສະຫຼາດສະຫຼຽວດ້ວຍວິທີການແລະລະບົບພິເສດໃນທຸກໂຮງຮຽນ, ວິທະຍາໄລແລະມະຫາວິທະຍາໄລ.

ຈາກທັດສະນະທາງດ້ານຈິດໃຈ, ມັນເປັນເລື່ອງທີ່ບໍ່ມີເຫດຜົນທີ່ຈະສຶກສາສະເພາະບຸກຄະລິກກະພາບທາງປັນຍາ. ມະນຸດມີສາມບຸກຄະລິກກະພາບທີ່ຕ້ອງການການສຶກສາພື້ນຖານຢ່າງຮີບດ່ວນ.
