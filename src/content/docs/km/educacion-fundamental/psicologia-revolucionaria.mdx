គ្រូបង្រៀននៃសាលារៀន មហាវិទ្យាល័យ និងសាកលវិទ្យាល័យ គួរតែសិក្សាឱ្យបានស៊ីជម្រៅអំពីចិត្តវិទ្យាបដិវត្តន៍ដែលចលនាញាណនិយមអន្តរជាតិបង្រៀន។

ចិត្តវិទ្យានៃបដិវត្តន៍ដែលកំពុងដំណើរការគឺខុសគ្នាទាំងស្រុងពីអ្វីៗទាំងអស់ដែលធ្លាប់ស្គាល់ពីមុនមកដោយឈ្មោះនេះ។

ហួសពីការសង្ស័យ យើងអាចនិយាយបានដោយមិនខ្លាចខុសថា ក្នុងរយៈពេលរាប់សតវត្សរ៍ដែលបានកន្លងផុតទៅ ពីភាពងងឹតនៃគ្រប់វ័យ ចិត្តវិទ្យាមិនដែលធ្លាក់ចុះទាបដូចសព្វថ្ងៃនេះទេ នៅក្នុងយុគសម័យ "អ្នកបះបោរគ្មានហេតុផល" និងសុភាពបុរសរ៉ុក។

ចិត្តវិទ្យាថយក្រោយ និងប្រតិកម្មនៃសម័យទំនើបនេះកាន់តែអាក្រក់ទៅៗ បានបាត់បង់នូវហេតុផលនៃការមាន ហើយការទាក់ទងដោយផ្ទាល់ជាមួយប្រភពដើមពិតរបស់វា។

នៅក្នុងសម័យនៃការខូចទ្រង់ទ្រាយផ្លូវភេទ និងការខ្សោះជីវជាតិនៃចិត្តទាំងស្រុង វាមិនត្រឹមតែមិនអាចកំណត់ពាក្យ ចិត្តវិទ្យា ជាមួយនឹងភាពត្រឹមត្រូវដ៏ខ្សោយប៉ុណ្ណោះទេ ប៉ុន្តែលើសពីនេះទៅទៀត វាមិនស្គាល់នូវប្រធានបទគ្រឹះនៃចិត្តវិទ្យាពិតប្រាកដនោះទេ។

អ្នកដែលសន្មត់ខុសថា ចិត្តវិទ្យា គឺជាវិទ្យាសាស្ត្រសហសម័យនៃម៉ោងចុងក្រោយ ពិតជាមានការភ័ន្តច្រឡំ ព្រោះថា ចិត្តវិទ្យា គឺជាវិទ្យាសាស្ត្របុរាណដែល មានប្រភពដើមនៅក្នុងសាលាចាស់នៃអាថ៌កំបាំងបុរាណ។

ចំពោះប្រភេទនៃ SNOB, ចោរទំនើប, ភាពយឺតយ៉ាវ វាជាការមិនអាចទៅរួចទេក្នុងការកំណត់អ្វីដែលគេហៅថា ចិត្តវិទ្យា ព្រោះលើកលែងតែសម័យបច្ចុប្បន្ននេះ វាច្បាស់ណាស់ថា ចិត្តវិទ្យាមិនដែលបានកើតមានក្រោមឈ្មោះរបស់ខ្លួនទេ ដោយសារតែហេតុផលបែបនេះ ឬហេតុផលផ្សេងទៀត តែងតែសង្ស័យថាមានទំនោរបំផ្លិចបំផ្លាញនៃលក្ខណៈនយោបាយ ឬសាសនា ដូច្នេះហើយបានជាឃើញខ្លួនឯងត្រូវបិទបាំងខ្លួនឯងជាមួយនឹងសម្លៀកបំពាក់ជាច្រើន។

តាំងពីសម័យបុរាណមក នៅក្នុងឆាកផ្សេងៗគ្នានៃល្ខោនជីវិត ចិត្តវិទ្យាតែងតែដើរតួនាទីរបស់ខ្លួន ដោយក្លែងខ្លួនយ៉ាងឆ្លាតវៃជាមួយនឹងសម្លៀកបំពាក់នៃទស្សនវិជ្ជា។

នៅលើច្រាំងទន្លេគង្គា នៅក្នុងប្រទេសឥណ្ឌាដ៏ពិសិដ្ឋនៃវេដាស ពីភាពភ័យរន្ធត់នៃសតវត្សរ៍មក មានទម្រង់នៃយូហ្គា ដែលនៅក្នុងជម្រៅ គឺសុទ្ធសាធជាចិត្តវិទ្យាពិសោធន៍នៃជើងហោះហើរខ្ពស់។

យូហ្គាទាំងប្រាំពីរ តែងតែត្រូវបានពិពណ៌នាថាជាវិធីសាស្ត្រ នីតិវិធី ឬប្រព័ន្ធទស្សនវិជ្ជា។

នៅក្នុងពិភពអារ៉ាប់ ការបង្រៀនដ៏ពិសិដ្ឋរបស់ពួកស៊ូហ្វី ដែលជាផ្នែកមួយនៃមេតាហ្វីសិច ជាផ្នែកមួយនៃសាសនា ពិតជាមានលំដាប់ផ្លូវចិត្តទាំងស្រុង។

នៅក្នុងទ្វីបអឺរ៉ុបចាស់ ដែលរលួយដល់ខួរឆ្អឹងជាមួយនឹងសង្គ្រាម ការរើសអើងពូជសាសន៍ សាសនា នយោបាយ ជាដើម រហូតដល់ចុងសតវត្សចុងក្រោយនេះ ចិត្តវិទ្យាបានក្លែងខ្លួនជាមួយនឹងឈុតនៃទស្សនវិជ្ជាដើម្បីឱ្យអាចមើលរំលងបាន។

ទស្សនវិជ្ជា ទោះបីជាមានការបែងចែក និងការបែងចែករងទាំងអស់របស់វា ដូចជាតក្កវិជ្ជា ទ្រឹស្តីនៃចំណេះដឹង សីលធម៌ សោភ័ណភាព ជាដើម ក៏ដោយ គឺជាការឆ្លុះបញ្ចាំងដោយខ្លួនឯងជាក់ស្តែង ចំណេះដឹងអាថ៌កំបាំងនៃសត្វ មុខងារនៃការយល់ដឹងនៃមនសិការភ្ញាក់ ហួសពីការសង្ស័យណាមួយ។

កំហុសនៃសាលាទស្សនវិជ្ជាជាច្រើនស្ថិតនៅលើការចាត់ទុកចិត្តវិទ្យាថាជាអ្វីដែលទាបជាងទស្សនវិជ្ជា ជាអ្វីមួយដែលទាក់ទងតែនឹងទិដ្ឋភាពទាបបំផុត និងសូម្បីតែរឿងតូចតាចនៃធម្មជាតិរបស់មនុស្ស។

ការសិក្សាប្រៀបធៀបនៃសាសនាអនុញ្ញាតឱ្យយើងឈានទៅដល់ការសន្និដ្ឋានឡូជីខលថា វិទ្យាសាស្ត្រនៃចិត្តវិទ្យា តែងតែមានទំនាក់ទំនងយ៉ាងស្និទ្ធស្នាលជាមួយនឹងគោលការណ៍សាសនាទាំងអស់។ ការសិក្សាប្រៀបធៀបណាមួយនៃសាសនាមកបង្ហាញយើងថានៅក្នុងអក្សរសិល្ប៍ដ៏ពិសិដ្ឋបំផុតនៃប្រទេសផ្សេងៗគ្នា និងសម័យផ្សេងៗគ្នា មានកំណប់ទ្រព្យដ៏អស្ចារ្យនៃវិទ្យាសាស្ត្រផ្លូវចិត្ត។

ការស៊ើបអង្កេតស៊ីជម្រៅក្នុងវិស័យនៃ GNOSTICISM អនុញ្ញាតឱ្យយើងស្វែងរកការចងក្រងដ៏អស្ចារ្យនោះនៃអ្នកនិពន្ធញាណនិយមជាច្រើនដែលមកពីសម័យដំបូងនៃសាសនាគ្រឹស្ត ហើយត្រូវបានគេស្គាល់ក្រោមចំណងជើង PHILOKALIA ដែលនៅតែប្រើសព្វថ្ងៃនេះនៅក្នុងព្រះវិហារបូព៌ា ជាពិសេសសម្រាប់ការបង្រៀនដល់ព្រះសង្ឃ។

ហួសពីការសង្ស័យ ហើយដោយគ្មានការភ័យខ្លាចតិចតួចបំផុតក្នុងការធ្លាក់ចូលទៅក្នុងភាពបោកប្រាស់ យើងអាចអះអាងយ៉ាងម៉ឺងម៉ាត់ថា PHILOKALIA គឺជាចិត្តវិទ្យាពិសោធន៍សុទ្ធសាធដោយសារធាតុ។

នៅក្នុងសាលាអាថ៌កំបាំងបុរាណនៃប្រទេសក្រិក អេហ្ស៊ីប រ៉ូម ឥណ្ឌា ពែរ្ស ម៉ិកស៊ិក ប៉េរូ អាសស៊ើរ ខាល់ឌា ជាដើម ចិត្តវិទ្យាតែងតែភ្ជាប់ជាមួយទស្សនវិជ្ជា សិល្បៈគោលបំណងពិតប្រាកដ វិទ្យាសាស្ត្រ និងសាសនា។

នៅសម័យបុរាណ ចិត្តវិទ្យាត្រូវបានលាក់បាំងយ៉ាងឆ្លាតវៃក្នុងចំណោមទម្រង់ដ៏ស្រស់ស្អាតរបស់អ្នករាំដ៏ពិសិដ្ឋ ឬក្នុងចំណោមអាថ៌កំបាំងនៃអក្សរសិល្ប៍ចម្លែក ឬរូបចម្លាក់ដ៏ស្រស់ស្អាត ឬនៅក្នុងកំណាព្យ ឬនៅក្នុងសោកនាដកម្ម និងសូម្បីតែនៅក្នុងតន្ត្រីដ៏រីករាយនៃប្រាសាទ។

មុនពេលវិទ្យាសាស្ត្រ ទស្សនវិជ្ជា សិល្បៈ និងសាសនាបានបែកគ្នាដើម្បីបង្វែរដោយឯករាជ្យ ចិត្តវិទ្យាបានសោយរាជ្យជាអធិបតីនៅក្នុងសាលាអាថ៌កំបាំងចាស់បំផុតទាំងអស់។

នៅពេលដែលមហាវិទ្យាល័យផ្តួចផ្តើមត្រូវបានបិទដោយសារតែ KALIYUGA ឬ AGE BLACK ដែលយើងនៅតែស្ថិតក្នុងនោះ ចិត្តវិទ្យាបានរស់រានមានជីវិតក្នុងចំណោមនិមិត្តសញ្ញានៃសាលា ESOTERIC និង PSEUDO-ESOTERIC ផ្សេងៗគ្នានៃពិភពលោកសម័យទំនើប ហើយជាពិសេសក្នុងចំណោម ESOTERICISM ញាណនិយម។

ការវិភាគស៊ីជម្រៅ និងការស៊ើបអង្កេតស៊ីជម្រៅ អនុញ្ញាតឱ្យយើងយល់យ៉ាងច្បាស់ថា ប្រព័ន្ធ និងគោលលទ្ធិផ្លូវចិត្តផ្សេងៗ ដែលមាននៅក្នុងអតីតកាល និងដែលមាននៅក្នុងបច្ចុប្បន្ន អាចត្រូវបានបែងចែកជាពីរប្រភេទ។

ដំបូង - គោលលទ្ធិដូចដែលបញ្ញវន្តជាច្រើនសន្មត់។ ចិត្តវិទ្យាទំនើបជាការពិតជាកម្មសិទ្ធិរបស់ប្រភេទនេះ។

ទីពីរ - គោលលទ្ធិដែលសិក្សាពីមនុស្សពីទស្សនៈនៃបដិវត្តន៍នៃមនសិការ។

ក្រោយមកទៀតគឺជាការពិតដែលជាគោលលទ្ធិដើម គោលលទ្ធិចាស់ជាងគេ មានតែពួកគេប៉ុណ្ណោះដែលអនុញ្ញាតឱ្យយើងយល់ពីប្រភពដើមរស់រវើកនៃចិត្តវិទ្យានិងអត្ថន័យជ្រាលជ្រៅរបស់វា។

នៅពេលដែលយើងទាំងអស់គ្នាបានយល់យ៉ាងពេញលេញ និងគ្រប់កម្រិតនៃចិត្ត ថាតើការសិក្សាអំពីមនុស្សមានសារៈសំខាន់យ៉ាងណាចំពោះទស្សនៈថ្មីនៃបដិវត្តន៍នៃមនសិការ នោះយើងនឹងយល់ថា ចិត្តវិទ្យា គឺជាការសិក្សាអំពីគោលការណ៍ ច្បាប់ និងអង្គហេតុ ដែលទាក់ទងយ៉ាងស្និទ្ធស្នាលទៅនឹងការផ្លាស់ប្តូររ៉ាឌីកាល់ និងច្បាស់លាស់របស់បុគ្គល។

វាជាការបន្ទាន់ដែលគ្រូបង្រៀននៃសាលារៀន មហាវិទ្យាល័យ និងសាកលវិទ្យាល័យ យល់យ៉ាងពេញលេញអំពីម៉ោងដ៏សំខាន់ដែលយើងកំពុងរស់នៅ និងស្ថានភាពមហន្តរាយនៃការវង្វេងផ្លូវចិត្តដែលជំនាន់ថ្មីកំពុងប្រឈមមុខ។
