វាចាំបាច់ណាស់ដែលត្រូវស្រឡាញ់ និងត្រូវបានគេស្រឡាញ់ ប៉ុន្តែជាអកុសលសម្រាប់ពិភពលោក មនុស្សមិនស្រឡាញ់ ឬក៏មិនត្រូវបានគេស្រឡាញ់ឡើយ។

អ្វីដែលគេហៅថា សេចក្តីស្រឡាញ់ គឺជាអ្វីដែលមនុស្សមិនស្គាល់ ហើយគេច្រឡំយ៉ាងងាយស្រួលជាមួយនឹងតណ្ហា និងការភ័យខ្លាច។

ប្រសិនបើមនុស្សអាចស្រឡាញ់ និងត្រូវបានគេស្រឡាញ់ សង្គ្រាមនឹងមិនអាចកើតមានឡើងនៅលើផែនដីបានទេ។

អាពាហ៍ពិពាហ៍ជាច្រើនដែលអាចពិតជាមានសុភមង្គល បែរជាមិនមាន ដោយសារតែការអាក់អន់ចិត្តចាស់ៗដែលកកកុញនៅក្នុងការចងចាំ។

ប្រសិនបើប្តីប្រពន្ធមានចិត្តសប្បុរស ពួកគេនឹងបំភ្លេចអតីតកាលដ៏ឈឺចាប់ ហើយរស់នៅក្នុងភាពពេញលេញ ពោរពេញដោយសុភមង្គលពិតប្រាកដ។

ចិត្តសម្លាប់សេចក្តីស្រឡាញ់ បំផ្លាញវា។ បទពិសោធន៍ ការមិនសប្បាយចិត្តចាស់ៗ ការច្រណែនពីអតីតកាល អ្វីៗទាំងអស់នេះដែលកកកុញនៅក្នុងការចងចាំ បំផ្លាញសេចក្តីស្រឡាញ់។

ប្រពន្ធដែលអាក់អន់ចិត្តជាច្រើនអាចសប្បាយចិត្ត ប្រសិនបើពួកគេមានចិត្តសប្បុរសគ្រប់គ្រាន់ដើម្បីបំភ្លេចអតីតកាល ហើយរស់នៅក្នុងបច្ចុប្បន្ន ដោយស្រឡាញ់ប្តី។

ប្តីជាច្រើនអាចពិតជាសប្បាយចិត្តជាមួយប្រពន្ធរបស់ពួកគេ ប្រសិនបើពួកគេមានចិត្តសប្បុរសគ្រប់គ្រាន់ ដើម្បីអត់ទោសកំហុសចាស់ៗ ហើយបំភ្លេចចោលនូវជម្លោះ និងការខកចិត្តដែលកកកុញនៅក្នុងការចងចាំ។

វាចាំបាច់ណាស់ វាជារឿងបន្ទាន់ដែលអាពាហ៍ពិពាហ៍យល់ពីអត្ថន័យដ៏ជ្រាលជ្រៅនៃពេលនេះ។

ប្តីប្រពន្ធគួរតែមានអារម្មណ៍ដូចជាគូស្វាមីភរិយាថ្មីជានិច្ច ដោយបំភ្លេចអតីតកាល ហើយរស់នៅយ៉ាងសប្បាយរីករាយក្នុងពេលបច្ចុប្បន្ន។

សេចក្តីស្រឡាញ់ និងការអាក់អន់ចិត្ត គឺជាសារធាតុអាតូមដែលមិនត្រូវគ្នា។ នៅក្នុងសេចក្តីស្រឡាញ់មិនអាចមានការអាក់អន់ចិត្តប្រភេទណាមួយឡើយ។ សេចក្តីស្រឡាញ់គឺជាការអភ័យទោសដ៏អស់កល្ប។

មានសេចក្តីស្រឡាញ់ចំពោះអ្នកដែល មានទុក្ខព្រួយពិតប្រាកដចំពោះការរងទុក្ខរបស់មិត្តភក្តិ និងសត្រូវរបស់ពួកគេ។ មានសេចក្តីស្រឡាញ់ពិតប្រាកដនៅក្នុងអ្នកដែលធ្វើការដោយអស់ពីចិត្ត ដើម្បីសុខុមាលភាពនៃមនុស្សរាបសារ អ្នកក្រីក្រ អ្នកខ្វះខាត។

មានសេចក្តីស្រឡាញ់ចំពោះអ្នកដែលដោយឯកឯង និងធម្មជាតិ មានការអាណិតអាសូរចំពោះកសិករដែលស្រោចស្រពប្រឡាយដោយញើសរបស់គាត់ ចំពោះអ្នកភូមិដែលរងទុក្ខ ចំពោះអ្នកសុំទានដែលសុំកាក់ និងចំពោះឆ្កែដែលរាបសារដែលកំពុងព្រួយបារម្ភ និងឈឺ ដែលស្លាប់ដោយសារភាពអត់ឃ្លាននៅតាមដងផ្លូវ។

នៅពេលដែលយើងជួយនរណាម្នាក់ដោយអស់ពីចិត្ត នៅពេលដែលយើងថែរក្សាដើមឈើ និងស្រោចស្រពផ្កានៅក្នុងសួនច្បារដោយធម្មជាតិ និងឯកឯង ដោយគ្មាននរណាម្នាក់ទាមទារឱ្យយើងធ្វើដូច្នេះ មានចិត្តសប្បុរសពិតប្រាកដ ការអាណិតអាសូរពិតប្រាកដ សេចក្តីស្រឡាញ់ពិតប្រាកដ។

ជាអកុសលសម្រាប់ពិភពលោក មនុស្សមិនមានចិត្តសប្បុរសពិតប្រាកដទេ។ មនុស្សព្រួយបារម្ភតែចំពោះសមិទ្ធផលអាត្មានិយម បំណងប្រាថ្នា ភាពជោគជ័យ ចំណេះដឹង បទពិសោធន៍ ការរងទុក្ខ ភាពរីករាយ។ល។

នៅក្នុងពិភពលោកមានមនុស្សជាច្រើន ដែលមានតែចិត្តសប្បុរសក្លែងក្លាយប៉ុណ្ណោះ។ មានចិត្តសប្បុរសក្លែងក្លាយនៅក្នុងអ្នកនយោបាយដ៏ឈ្លាសវៃ នៅក្នុងកញ្ជ្រោងបោះឆ្នោតដែលខ្ជះខ្ជាយលុយកាក់ដោយមានគោលបំណងអាត្មានិយម ដើម្បីទទួលបានអំណាច កិត្តិយស ទីតាំង ទ្រព្យសម្បត្តិ។ល។ យើងមិនគួរច្រឡំឆ្មាជាមួយទន្សាយទេ។

ចិត្តសប្បុរសពិតប្រាកដគឺគ្មានប្រយោជន៍ទាំងស្រុង ប៉ុន្តែវាអាចត្រូវបានគេច្រឡំយ៉ាងងាយស្រួលជាមួយនឹងចិត្តសប្បុរសអាត្មានិយមក្លែងក្លាយ របស់កញ្ជ្រោងនយោបាយ ចោរ മുതலாភាព អ្នកដែលលោភលន់ស្ត្រី។ល។

យើងត្រូវតែមានចិត្តសប្បុរស។ ចិត្តសប្បុរសពិតប្រាកដ មិនមែនមកពីចិត្តទេ ចិត្តសប្បុរសពិតប្រាកដគឺជាក្លិនក្រអូបនៃបេះដូង។

ប្រសិនបើមនុស្សមានចិត្តសប្បុរស ពួកគេនឹងបំភ្លេចចោលនូវការអាក់អន់ចិត្តទាំងអស់ដែលកកកុញនៅក្នុងការចងចាំ បទពិសោធន៍ដ៏ឈឺចាប់ទាំងអស់នៃអតីតកាលជាច្រើន ហើយរៀនរស់នៅពីមួយពេលទៅមួយពេល ដោយតែងតែមានសុភមង្គល តែងតែមានចិត្តសប្បុរស ពោរពេញដោយភាពស្មោះត្រង់ពិតប្រាកដ។

ជាអកុសល អត្តា គឺជាការចងចាំ ហើយរស់នៅក្នុងអតីតកាល ចង់ត្រឡប់ទៅអតីតកាលជានិច្ច។ អតីតកាលបញ្ចប់ដោយមនុស្ស បំផ្លាញសុភមង្គល សម្លាប់សេចក្តីស្រឡាញ់។

ចិត្តដែលជាប់នៅក្នុងអតីតកាល មិនអាចយល់បានទាំងស្រុងនូវអត្ថន័យដ៏ជ្រាលជ្រៅនៃពេលដែលយើងកំពុងរស់នៅនោះទេ។

មានមនុស្សជាច្រើនដែលសរសេរមកយើងស្វែងរកការកំសាន្តចិត្ត សុំក្រែមលាបដ៏មានតម្លៃដើម្បីព្យាបាលបេះដូងដែលឈឺចាប់របស់ពួកគេ ប៉ុន្តែមានមនុស្សតិចណាស់ដែលព្រួយបារម្ភអំពីការកំសាន្តចិត្តអ្នកដែលសោកសៅ។

មានមនុស្សជាច្រើនសរសេរមកយើងដើម្បីរៀបរាប់អំពីស្ថានភាពដ៏វេទនារបស់ពួកគេ ប៉ុន្តែមានមនុស្សកម្រណាស់ដែលចែកនំប៉័ងតែមួយគត់ដែលត្រូវចិញ្ចឹមពួកគេដើម្បីចែករំលែកវាជាមួយអ្នកដែលខ្វះខាតផ្សេងទៀត។

មនុស្សមិនចង់យល់ថា នៅពីក្រោយរាល់ឥទ្ធិពលមានបុព្វហេតុ ហើយថាដោយគ្រាន់តែផ្លាស់ប្តូរបុព្វហេតុ យើងកែប្រែឥទ្ធិពល។
