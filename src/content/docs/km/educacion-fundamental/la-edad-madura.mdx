វ័យកណ្ដាលចាប់ផ្ដើមនៅអាយុសាមសិបប្រាំឆ្នាំ ហើយបញ្ចប់នៅអាយុហាសិបប្រាំមួយឆ្នាំ។

បុរសវ័យកណ្ដាលត្រូវចេះគ្រប់គ្រងផ្ទះ និងណែនាំកូនៗរបស់ខ្លួន។

នៅក្នុងជីវិតធម្មតា បុរសវ័យកណ្ដាលគ្រប់រូបគឺជាមេគ្រួសារ។ បុរសដែលមិនបានបង្កើតគ្រួសារ និងទ្រព្យសម្បត្តិរបស់ខ្លួនក្នុងវ័យក្មេង និងវ័យកណ្ដាល នោះគេលែងបានបង្កើតទៀតហើយ ពោលគឺគេជាមនុស្សបរាជ័យ។

អ្នកដែលព្យាយាមបង្កើតគ្រួសារ និងទ្រព្យសម្បត្តិក្នុងវ័យចាស់ជរា ពិតជាគួរឱ្យអាណិតណាស់។

សភាវគតិលោភលន់តែងតែហួសហេតុ និងចង់ប្រមូលទ្រព្យសម្បត្តិដ៏ច្រើនសន្ធឹកសន្ធាប់។ មនុស្សត្រូវការអាហារ សម្លៀកបំពាក់ និងទីជម្រក។ យើងត្រូវតែមានអាហារ ផ្ទះផ្ទាល់ខ្លួន សម្លៀកបំពាក់ ខោអាវ អាវរងារសម្រាប់គ្របដណ្ដប់រាងកាយ ប៉ុន្តែមិនចាំបាច់ប្រមូលលុយកាក់ដ៏ច្រើនសន្ធឹកសន្ធាប់ដើម្បីរស់នៅនោះទេ។

យើងមិនគាំទ្រភាពមានបាន ឬភាពក្រីក្រនោះទេ ព្រោះចំនុចខ្លាំងទាំងពីរនេះសុទ្ធតែអាចឱ្យគេតិះដៀលបាន។

មនុស្សជាច្រើនកំពុងតែរមៀលខ្លួននៅក្នុងភក់នៃភាពក្រីក្រ ហើយក៏មានមនុស្សមិនតិចដែរដែលកំពុងតែរមៀលខ្លួននៅក្នុងភក់នៃទ្រព្យសម្បត្តិ។

យើងត្រូវតែមានទ្រព្យសម្បត្តិសមរម្យ ពោលគឺផ្ទះដ៏ស្រស់ស្អាតដែលមានសួនច្បារដ៏ស្រស់បំព្រង ប្រភពចំណូលប្រកបដោយសុវត្ថិភាព តែងតែបង្ហាញខ្លួនឱ្យបានល្អ និងមិនត្រូវអត់ឃ្លានឡើយ។ នេះគឺជារឿងធម្មតាសម្រាប់មនុស្សគ្រប់រូប។

ភាពក្រីក្រ ភាពអត់ឃ្លាន ជំងឺ និងភាពល្ងង់ខ្លៅ មិនត្រូវកើតមាននៅក្នុងប្រទេសណាមួយដែលអះអាងថាខ្លួនជានរដ្ឋមានវប្បធម៌ និងអរិយធម៌នោះទេ។

លទ្ធិប្រជាធិបតេយ្យនៅមិនទាន់មាននៅឡើយទេ ប៉ុន្តែយើងត្រូវតែបង្កើតវា។ ដរាបណាមានពលរដ្ឋតែម្នាក់ដែលគ្មានអាហារ សម្លៀកបំពាក់ និងទីជម្រក នោះលទ្ធិប្រជាធិបតេយ្យពិតជាគ្រាន់តែជាឧត្ដមគតិដ៏ស្រស់ស្អាតប៉ុណ្ណោះ។

មេគ្រួសារត្រូវតែមានការយោគយល់ ឆ្លាតវៃ មិនត្រូវផឹកស្រា ស្រវឹង ហូបច្រើនពេក ឬជាមនុស្សឃោរឃៅជាដើមឡើយ។

បុរសពេញវ័យគ្រប់រូបដឹងតាមរយៈបទពិសោធន៍ផ្ទាល់ខ្លួនថា កូនៗធ្វើតាមគំរូរបស់ខ្លួន ហើយប្រសិនបើគំរូនោះខុស នោះវានឹងកំណត់ផ្លូវខុសឆ្គងដល់កូនចៅរបស់ខ្លួន។

ពិតជាល្ងង់ខ្លៅណាស់ដែលបុរសពេញវ័យមានប្រពន្ធច្រើននាក់ និងរស់នៅក្នុងភាពស្រវឹង ពិធីជប់លៀង ពិធីផឹកស៊ីសប្បាយជាដើម។

បុរសពេញវ័យមានទំនួលខុសត្រូវចំពោះគ្រួសារទាំងមូល ហើយវាច្បាស់ណាស់ថា ប្រសិនបើគាត់ដើរលើផ្លូវខុស នោះគាត់នឹងនាំមកនូវភាពច្របូកច្របល់កាន់តែច្រើនដល់ពិភពលោក ភាពច្របូកច្របល់កាន់តែច្រើន និងភាពជូរចត់កាន់តែច្រើន។

ឪពុក និងម្ដាយត្រូវតែយល់ពីភាពខុសគ្នារវាងភេទ។ វាជារឿងមិនសមហេតុផលដែលកូនស្រីសិក្សាពី រូបវិទ្យា គីមីវិទ្យា ពិជគណិត ជាដើម។ ខួរក្បាលរបស់ស្ត្រីខុសពីបុរស មុខវិជ្ជាទាំងនេះសាកសមនឹងបុរស ប៉ុន្តែគ្មានប្រយោជន៍ ហើយថែមទាំងបង្កគ្រោះថ្នាក់ដល់ចិត្តគំនិតស្ត្រីទៀតផង។

ឪពុកម្ដាយត្រូវតែខិតខំប្រឹងប្រែងអស់ពីចិត្តដើម្បីជំរុញការផ្លាស់ប្ដូរដ៏សំខាន់នៅក្នុងផែនការសិក្សារបស់សាលាទាំងអស់។

ស្ត្រីត្រូវតែរៀនអាន សរសេរ លេងព្យាណូ ត្បាញ ប៉ាក់ និងជាទូទៅ ជំនាញស្ត្រីគ្រប់ប្រភេទ។

ស្ត្រីត្រូវតែត្រៀមខ្លួនតាំងពីកៅអីសាលាសម្រាប់បេសកកម្មដ៏ឧត្ដុង្គឧត្ដមដែលជាកាតព្វកិច្ចរបស់ខ្លួនក្នុងនាមជាម្ដាយ និងជាភរិយា។

វាជារឿងមិនសមហេតុផលទេក្នុងការបំផ្លាញខួរក្បាលរបស់ស្ត្រីជាមួយនឹងការសិក្សាដ៏ស្មុគស្មាញ និងពិបាកដែលសាកសមសម្រាប់បុរស។

ឪពុកម្ដាយ និងគ្រូបង្រៀននៅសាលារៀន មហាវិទ្យាល័យ និងសាកលវិទ្យាល័យ ត្រូវតែយកចិត្តទុកដាក់បន្ថែមទៀតក្នុងការនាំស្ត្រីឱ្យក្លាយជាស្ត្រីភាពដែលខ្លួនគួរតែមាន។ វាជារឿងល្ងង់ខ្លៅក្នុងការធ្វើឱ្យស្ត្រីក្លាយជាយោធិន ដោយបង្ខំពួកគេឱ្យដើរដង្ហែក្បួនជាមួយនឹងទង់ជាតិ និងស្គរតាមដងផ្លូវនៃទីក្រុង ដូចជាពួកគេជាបុរស។

ស្ត្រីត្រូវតែជាស្ត្រីពិតប្រាកដ ហើយបុរសត្រូវតែជាបុរសពិតប្រាកដ។

ភេទកណ្ដាល គឺភាពស្រឡាញ់ភេទដូចគ្នា គឺជាផលិតផលនៃការខូចទ្រង់ទ្រាយ និងភាពព្រៃផ្សៃ។

នារីដែលខិតខំសិក្សាដ៏វែងឆ្ងាយ និងពិបាក នឹងក្លាយជាមនុស្សចាស់ ហើយគ្មានអ្នកណាម្នាក់រៀបការជាមួយពួកគេទេ។

នៅក្នុងជីវិតសម័យទំនើប ស្ត្រីគួរតែធ្វើការងាររយៈពេលខ្លី ដូចជា វប្បធម៌សម្រស់ វាយអក្សរ សរសេរអក្សរកាត់ កាត់ដេរ គរុកោសល្យ ជាដើម។

ជាធម្មតា ស្ត្រីគួរតែលះបង់ខ្លួនតែចំពោះជីវិតគ្រួសារប៉ុណ្ណោះ ប៉ុន្តែដោយសារតែភាពឃោរឃៅនៃសម័យកាលដែលយើងកំពុងរស់នៅ ស្ត្រីត្រូវការធ្វើការដើម្បីចិញ្ចឹមជីវិត និងរស់នៅ។

នៅក្នុងសង្គមដែលមានវប្បធម៌ និងអរិយធម៌ពិតប្រាកដ ស្ត្រីមិនចាំបាច់ធ្វើការនៅខាងក្រៅផ្ទះដើម្បីរស់នៅនោះទេ។ ការធ្វើការនៅខាងក្រៅផ្ទះនេះ គឺជាភាពឃោរឃៅដ៏អាក្រក់បំផុត។

បុរសសព្វថ្ងៃដែលខូចទ្រង់ទ្រាយបានបង្កើតសណ្ដាប់ធ្នាប់មិនពិត និងបានធ្វើឱ្យស្ត្រីបាត់បង់ភាពជាស្ត្រី បានដកស្ត្រីចេញពីផ្ទះរបស់ខ្លួន ហើយប្រែក្លាយពួកគេទៅជាទាសករ។
