វ័យកណ្តាលចាប់ផ្តើមនៅអាយុសាមសិបប្រាំឆ្នាំ ហើយបញ្ចប់នៅអាយុហាសិបប្រាំមួយឆ្នាំ។

បុរសវ័យកណ្តាលត្រូវចេះគ្រប់គ្រងផ្ទះសម្បែង និងណែនាំកូនចៅ។

នៅក្នុងជីវិតធម្មតា បុរសវ័យកណ្តាលគ្រប់រូបគឺជាមេគ្រួសារ។ បុរសដែលមិនបានបង្កើតគ្រួសារ និងទ្រព្យសម្បត្តិក្នុងវ័យក្មេង និងវ័យកណ្តាល លែងអាចបង្កើតបានទៀតហើយ គឺពិតជាបរាជ័យមែន។

អស់អ្នកដែលព្យាយាមបង្កើតគ្រួសារ និងទ្រព្យសម្បត្តិក្នុងវ័យចាស់ពិតជាគួរឱ្យអាណិតណាស់។

ភាពអាត្មានិយមនៃសេចក្តីលោភលន់ឈានដល់កម្រិតខ្លាំង ហើយចង់ប្រមូលទ្រព្យសម្បត្តិដ៏ច្រើនសន្ធឹកសន្ធាប់។ មនុស្សត្រូវការអាហារ សម្លៀកបំពាក់ និងជម្រក។ ចាំបាច់ត្រូវមានអាហារ ផ្ទះផ្ទាល់ខ្លួន សម្លៀកបំពាក់ ឈុត និងអាវដើម្បីគ្របដណ្តប់រាងកាយ ប៉ុន្តែមិនចាំបាច់ប្រមូលលុយយ៉ាងច្រើនដើម្បីរស់នៅនោះទេ។

យើងមិនការពារទ្រព្យសម្បត្តិ ឬភាពក្រីក្រនោះទេ ព្រោះចុងទាំងពីរនេះសុទ្ធតែត្រូវថ្កោលទោស។

មានមនុស្សជាច្រើនកំពុងតែរស់នៅក្នុងភក់នៃភាពក្រីក្រ ហើយក៏មានមនុស្សជាច្រើនកំពុងតែរស់នៅក្នុងភក់នៃទ្រព្យសម្បត្តិផងដែរ។

វាចាំបាច់ណាស់ក្នុងការមានទ្រព្យសម្បត្តិសមរម្យ នោះគឺផ្ទះដ៏ស្រស់ស្អាតជាមួយនឹងសួនច្បារដ៏ស្រស់ស្អាត ប្រភពចំណូលដែលមានសុវត្ថិភាព តែងតែបង្ហាញខ្លួនឱ្យបានល្អ និងមិនអត់ឃ្លាន។ នេះជារឿងធម្មតាសម្រាប់មនុស្សគ្រប់រូប។

ភាពក្រីក្រ ភាពអត់ឃ្លាន ជំងឺ និងភាពល្ងង់ខ្លៅ មិនគួរកើតមាននៅក្នុងប្រទេសណាមួយដែលអះអាងថាមានវប្បធម៌ និងអរិយធម៌នោះទេ។

លទ្ធិប្រជាធិបតេយ្យនៅមិនទាន់មាននៅឡើយទេ ប៉ុន្តែយើងត្រូវបង្កើតវា។ ដរាបណាមានពលរដ្ឋតែម្នាក់គត់ដែលគ្មានអាហារ សម្លៀកបំពាក់ និងជម្រក នោះលទ្ធិប្រជាធិបតេយ្យជាក់ស្តែងគ្រាន់តែជាឧត្តមគតិដ៏ស្រស់ស្អាតប៉ុណ្ណោះ។

មេគ្រួសារត្រូវមានការយល់ចិត្ត ឆ្លាតវៃ មិនដែលផឹកស្រា ចូលចិត្តស៊ីផឹក ស្រវឹង ផ្តាច់ការជាដើម។

បុរសពេញវ័យគ្រប់រូបដឹងតាមរយៈបទពិសោធន៍ផ្ទាល់ខ្លួនថាកូនចៅយកតម្រាប់តាមគំរូរបស់ខ្លួន ហើយប្រសិនបើគំរូនោះខុស នោះវានឹងកំណត់ផ្លូវដែលមិនសមហេតុផលសម្រាប់កូនចៅរបស់ពួកគេ។

វាពិតជាឆោតល្ងង់ណាស់ដែលបុរសពេញវ័យមានប្រពន្ធច្រើននាក់ ហើយរស់នៅក្នុងភាពស្រវឹង ពិធីជប់លៀង និងការសប្បាយហួសហេតុ។

បុរសពេញវ័យមានទំនួលខុសត្រូវចំពោះគ្រួសារទាំងមូល ហើយវាច្បាស់ណាស់ថា ប្រសិនបើគាត់ដើរលើផ្លូវខុស គាត់នឹងនាំមកនូវភាពវឹកវរ ការភ័ន្តច្រឡំ និងភាពជូរចត់កាន់តែច្រើនដល់ពិភពលោក។

ឪពុកនិងម្តាយត្រូវយល់ពីភាពខុសគ្នារវាងភេទ។ វាជារឿងមិនសមហេតុផលដែលកូនស្រីរៀនរូបវិទ្យា គីមីវិទ្យា ពិជគណិតជាដើម។ ខួរក្បាលរបស់ស្ត្រីខុសពីខួរក្បាលរបស់បុរស មុខវិជ្ជាទាំងនោះសាកសមនឹងបុរស ប៉ុន្តែគ្មានប្រយោជន៍ ហើយថែមទាំងបង្កគ្រោះថ្នាក់ដល់ខួរក្បាលស្ត្រីទៀតផង។

ឪពុកម្តាយត្រូវខិតខំប្រឹងប្រែងអស់ពីចិត្តដើម្បីជំរុញការផ្លាស់ប្តូរដ៏សំខាន់នៅក្នុងផែនការសិក្សារបស់សាលាទាំងមូល។

ស្ត្រីគួរតែរៀនអាន សរសេរ លេងព្យាណូ ត្បាញ ប៉ាក់ និងជាទូទៅការងារស្ត្រីគ្រប់ប្រភេទ។

ស្ត្រីត្រូវតែត្រូវបានរៀបចំតាំងពីកៅអីសាលាសម្រាប់បេសកកម្មដ៏ឧត្តុង្គឧត្តមដែលត្រូវនឹងនាងជា ម្តាយ និងជាភរិយា។

វាជារឿងមិនសមហេតុផលទេក្នុងការធ្វើឱ្យខូចខួរក្បាលស្ត្រីជាមួយនឹងការសិក្សាដ៏ស្មុគស្មាញ និងលំបាក ដែលសាកសមនឹងបុរស។

វាចាំបាច់ណាស់ដែលទាំងឪពុកម្តាយ និងគ្រូបង្រៀននៅសាលារៀន មហាវិទ្យាល័យ និងសាកលវិទ្យាល័យយកចិត្តទុកដាក់បន្ថែមទៀតចំពោះការនាំស្ត្រីទៅរកភាពជាស្ត្រីដែលត្រូវនឹងនាង។ វាជារឿងឆោតល្ងង់ក្នុងការយោធូបនីយកម្មស្ត្រី បង្ខំពួកគេឱ្យដើរដង្ហែជាមួយទង់ជាតិ និងស្គរតាមដងផ្លូវនៃទីក្រុងដូចជាពួកគេជាបុរស។

ស្ត្រីគួរតែមានភាពជាស្ត្រី និងបុរសគួរតែមានភាពជាបុរស។

ភេទកណ្តាល ភាពស្រឡាញ់ភេទដូចគ្នា គឺជាផលិតផលនៃការខូចទ្រង់ទ្រាយ និងភាពព្រៃផ្សៃ។

នារីដែលលះបង់ខ្លួនឯងក្នុងការសិក្សាដ៏យូរ និងពិបាកក្លាយជាចាស់ ហើយគ្មាននរណាម្នាក់រៀបការជាមួយពួកគេនោះទេ។

នៅក្នុងជីវិតសម័យទំនើប វាជាការងាយស្រួលសម្រាប់ស្ត្រីក្នុងការធ្វើអាជីពខ្លីៗ វប្បធម៌នៃភាពស្រស់ស្អាត ការវាយអក្សរ ការសរសេរអក្សរកាត់ ការដេរ សិក្ខាសាលាជាដើម។

ជាធម្មតាស្ត្រីគួរតែលះបង់ខ្លួនឯងចំពោះជីវិតគ្រួសារ ប៉ុន្តែដោយសារតែភាពឃោរឃៅនៃសម័យកាលដែលយើងកំពុងរស់នៅ ស្ត្រីត្រូវធ្វើការដើម្បីចិញ្ចឹមជីវិត និងរស់នៅ។

នៅក្នុងសង្គមដែលមានវប្បធម៌ និងអរិយធម៌ពិតប្រាកដ ស្ត្រីមិនចាំបាច់ធ្វើការនៅខាងក្រៅផ្ទះដើម្បីរស់នៅនោះទេ។ ការធ្វើការនៅខាងក្រៅផ្ទះគឺជាភាពឃោរឃៅនៃប្រភេទដ៏អាក្រក់បំផុត។

បុរសសម័យទំនើបដែលខូចទ្រង់ទ្រាយបានបង្កើតនូវសណ្តាប់ធ្នាប់មិនពិត ហើយបានធ្វើឱ្យស្ត្រីបាត់បង់ភាពជាស្ត្រីរបស់ខ្លួន នាំនាងចេញពីផ្ទះរបស់នាង ហើយប្រែក្លាយនាងទៅជាទាសករ។
