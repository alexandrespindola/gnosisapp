យើងមិនអាចបដិសេធច្បាប់នៃការកើតឡើងដដែលៗ ដែលកំពុងដំណើរការគ្រប់ពេលវេលានៃជីវិតរបស់យើងបានឡើយ។

ពិតណាស់ ជារៀងរាល់ថ្ងៃនៃអត្ថិភាពរបស់យើង មានការកើតឡើងដដែលៗនៃព្រឹត្តិការណ៍ ស្ថានភាពនៃស្មារតី ពាក្យសម្ដី បំណងប្រាថ្នា គំនិត ចេតនាជាដើម។

វាច្បាស់ណាស់ថា នៅពេលដែលមនុស្សម្នាក់មិនសង្កេតមើលខ្លួនឯង នោះគេមិនអាចដឹងពីការកើតឡើងដដែលៗជារៀងរាល់ថ្ងៃនេះទេ។

វាជាភស្តុតាងដែលថា អ្នកដែលមិនមានចំណាប់អារម្មណ៍ក្នុងការសង្កេតមើលខ្លួនឯង ក៏មិនចង់ធ្វើការដើម្បីសម្រេចបាននូវការផ្លាស់ប្តូររ៉ាឌីកាល់ពិតប្រាកដដែរ។

អ្វីដែលអាក្រក់បំផុតនោះគឺ មានមនុស្សដែលចង់ផ្លាស់ប្តូរខ្លួនឯងដោយមិនធ្វើការលើខ្លួនឯង។

យើងមិនបដិសេធការពិតដែលថា មនុស្សគ្រប់រូបមានសិទ្ធិទទួលបានសុភមង្គលពិតប្រាកដនៃស្មារតីនោះទេ ប៉ុន្តែវាក៏ជាការពិតផងដែរថា សុភមង្គលនឹងក្លាយជារឿងដែលមិនអាចទៅរួច ប្រសិនបើយើងមិនធ្វើការលើខ្លួនឯង។

មនុស្សម្នាក់អាចផ្លាស់ប្តូរខាងក្នុងបាន នៅពេលដែលពួកគេពិតជាអាចកែប្រែប្រតិកម្មរបស់ពួកគេចំពោះព្រឹត្តិការណ៍ផ្សេងៗដែលកើតឡើងចំពោះពួកគេជារៀងរាល់ថ្ងៃ។

ទោះយ៉ាងណាក៏ដោយ យើងមិនអាចកែប្រែវិធីដែលយើងប្រតិកម្មចំពោះព្រឹត្តិការណ៍នៃជីវិតជាក់ស្តែងបានទេ ប្រសិនបើយើងមិនធ្វើការយ៉ាងម៉ត់ចត់លើខ្លួនឯង។

យើងត្រូវផ្លាស់ប្តូររបៀបគិតរបស់យើង ត្រូវមានការធ្វេសប្រហែសតិច ត្រូវមានភាពធ្ងន់ធ្ងរជាងមុន និងយកជីវិតក្នុងទម្រង់ខុសគ្នា ក្នុងន័យពិតប្រាកដ និងជាក់ស្តែងរបស់វា។

ទោះយ៉ាងណាក៏ដោយ ប្រសិនបើយើងបន្តដូចដែលយើងធ្លាប់ធ្វើ ដោយប្រព្រឹត្តតាមរបៀបដដែលៗជារៀងរាល់ថ្ងៃ ដោយធ្វើកំហុសដដែលៗ ដោយការធ្វេសប្រហែសដូចសព្វដង នោះលទ្ធភាពនៃការផ្លាស់ប្តូរណាមួយនឹងត្រូវលុបចោលជាក់ស្តែង។

ប្រសិនបើមនុស្សម្នាក់ពិតជាចង់ស្គាល់ខ្លួនឯង ពួកគេត្រូវតែចាប់ផ្តើមដោយការសង្កេតមើលអាកប្បកិរិយាផ្ទាល់ខ្លួនរបស់ពួកគេ ចំពោះព្រឹត្តិការណ៍ណាមួយនៃជីវិតប្រចាំថ្ងៃ។

យើងមិនមានន័យថា មនុស្សម្នាក់មិនគួរត្រួតពិនិត្យមើលខ្លួនឯងជារៀងរាល់ថ្ងៃនោះទេ យើងគ្រាន់តែចង់បញ្ជាក់ថា ពួកគេគួរតែចាប់ផ្តើមដោយការសង្កេតមើលថ្ងៃដំបូង។

អ្វីៗទាំងអស់ត្រូវតែមានការចាប់ផ្តើម ហើយការចាប់ផ្តើមដោយការសង្កេតមើលអាកប្បកិរិយារបស់យើងនៅថ្ងៃណាមួយនៃជីវិតរបស់យើង គឺជាការចាប់ផ្តើមដ៏ល្អ។

ការសង្កេតមើលប្រតិកម្មមេកានិចរបស់យើងចំពោះព័ត៌មានលម្អិតតូចតាចទាំងអស់នោះ នៅក្នុងបន្ទប់គេង ផ្ទះបាយ បន្ទប់បរិភោគអាហារ ផ្ទះ តាមផ្លូវ កន្លែងធ្វើការ។ល។ អ្វីដែលមនុស្សម្នាក់និយាយ អារម្មណ៍ និងគិត គឺពិតជាអ្វីដែលសមស្របបំផុត។

អ្វីដែលសំខាន់គឺត្រូវមើលពីរបៀប ឬតាមវិធីណាដែលមនុស្សម្នាក់អាចផ្លាស់ប្តូរប្រតិកម្មទាំងនោះ។ ទោះយ៉ាងណាក៏ដោយ ប្រសិនបើយើងជឿថាយើងជាមនុស្សល្អ ដែលយើងមិនដែលប្រព្រឹត្តដោយអចេតនា និងខុសនោះទេ យើងនឹងមិនផ្លាស់ប្តូរឡើយ។

ជាដំបូង យើងត្រូវយល់ថាយើងជាមនុស្ស-ម៉ាស៊ីន អាយ៉ងសាមញ្ញដែលគ្រប់គ្រងដោយភ្នាក់ងារសម្ងាត់ ដោយខ្លួនលាក់កំបាំង។

មានមនុស្សជាច្រើនរស់នៅក្នុងមនុស្សរបស់យើង យើងមិនដែលដូចគ្នាទេ។ ជួនកាលមនុស្សអាក្រក់ម្នាក់លេចឡើងនៅក្នុងខ្លួនយើង ពេលខ្លះមនុស្សឆាប់ខឹង នៅពេលផ្សេងទៀតមនុស្សអស្ចារ្យ មេត្តាករុណា ក្រោយមកមនុស្សអាស្រូវ ឬបង្កាច់បង្ខូច បន្ទាប់មកជាមនុស្សបរិសុទ្ធ បន្ទាប់មកជាអ្នកកុហក។ល។

យើងមានមនុស្សគ្រប់ប្រភេទនៅក្នុងខ្លួនយើងម្នាក់ៗ មានខ្ញុំគ្រប់ប្រភេទ។ បុគ្គលិកលក្ខណៈរបស់យើងមិនមែនជាអ្វីក្រៅពីអាយ៉ង តុក្កតាដែលចេះនិយាយ អ្វីមួយដែលជាមេកានិចនោះទេ។

ចូរចាប់ផ្តើមដោយការប្រព្រឹត្តដោយដឹងខ្លួនក្នុងរយៈពេលខ្លីនៃថ្ងៃ។ យើងត្រូវឈប់ធ្វើជាម៉ាស៊ីនសាមញ្ញ ទោះបីជាក្នុងរយៈពេលខ្លីប៉ុន្មាននាទីក្នុងមួយថ្ងៃក៏ដោយ នេះនឹងជះឥទ្ធិពលយ៉ាងដាច់អហង្ការលើអត្ថិភាពរបស់យើង។

នៅពេលដែលយើងសង្កេតមើលខ្លួនឯង ហើយមិនធ្វើអ្វីដែលខ្ញុំនេះ ឬខ្ញុំនោះចង់បាន នោះច្បាស់ណាស់ថាយើងចាប់ផ្តើមឈប់ធ្វើជាម៉ាស៊ីន។

គ្រាន់តែមួយពេលដែលយើងដឹងខ្លួនគ្រប់គ្រាន់ ដើម្បីឈប់ធ្វើជាម៉ាស៊ីន ប្រសិនបើវាត្រូវបានធ្វើដោយស្ម័គ្រចិត្ត នោះជាធម្មតាផ្លាស់ប្តូររ៉ាឌីកាល់នូវកាលៈទេសៈមិនល្អជាច្រើន។

ជាអកុសល យើងរស់នៅក្នុងជីវិតមេកានិច ទម្លាប់ និងគ្មានន័យជារៀងរាល់ថ្ងៃ។ យើងធ្វើម្តងទៀតនូវព្រឹត្តិការណ៍ ទម្លាប់របស់យើងគឺដូចគ្នា យើងមិនដែលចង់កែប្រែវាទេ ពួកគេគឺជាផ្លូវដែកមេកានិចដែលរថភ្លើងនៃអត្ថិភាពដ៏កំសត់របស់យើងធ្វើចរាចរណ៍ ទោះយ៉ាងណាក៏ដោយ យើងគិតថាល្អបំផុតអំពីខ្លួនយើង...

គ្រប់ទីកន្លែងមាន "អ្នកភូតកុហក" ច្រើនក្រៃលែង អ្នកដែលជឿថាខ្លួនឯងជាព្រះ។ សត្វមេកានិច ប្រព្រឹត្តតាមទម្លាប់ តួអង្គមកពីភក់នៃផែនដី អាយ៉ងកំសត់ដែលផ្លាស់ទីដោយខ្ញុំផ្សេងៗ។ មនុស្សបែបនេះនឹងមិនធ្វើការលើខ្លួនឯងទេ។
