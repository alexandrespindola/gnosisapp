ចាប់ពីថ្ងៃទី ២០ ខែកុម្ភៈ ដល់ថ្ងៃទី ២១ ខែមីនា

យើងបានមកដល់រាត្រីមាតានៃលោហធាតុអេហ្ស៊ីប, ជម្រៅមហាសមុទ្រនៃ ត្រី, ភាពងងឹតគ្មានព្រំដែននៃលំហ អរូបីដាច់ខាត; ធាតុដំបូងនៃជ្រោះដែលអណ្ដើកយាមមាសនៃ RHIN ឬភ្លើងនៃការគិតដ៏ទេវភាព និង ហ្សែនស៊ីស។

ត្រី ត្រូវបានតំណាងដោយត្រីពីរយ៉ាងឈ្លាសវៃ; ត្រី, ត្រី, គឺជា SOMA នៃអាថ៌កំបាំងនៃ អ៊ីស៊ីស។ ត្រី គឺជានិមិត្តសញ្ញារស់រវើកនៃគ្រិស្តសាសនា ជីណូស្ទីក បុរាណ។

ត្រីទាំងពីរនៃ ត្រី ដែលភ្ជាប់ដោយសហសញ្ញាមានអត្ថន័យជ្រាលជ្រៅ ជីណូស្ទីក, តំណាងឱ្យព្រលឹងទាំងពីរនៃ ELOHIM បុរាណដែលលិចនៅក្នុងទឹកជ្រៅនៃ រាត្រីមាតា។

យើងបានពន្យល់រួចហើយនៅក្នុងជំពូកមុនៗថា ភាពស្និទ្ធស្នាល, អង្គ, ATMAN មានព្រលឹងពីរ: មួយជាស្រី, មួយទៀតជាប្រុស។

យើងបានពន្យល់រួចហើយថា ព្រលឹងខាងវិញ្ញាណ, BUDDHI, គឺស្រី។ យើងបាននិយាយរួចហើយ ហើយយើងនិយាយម្តងទៀតថា ព្រលឹងមនុស្ស, MANAS ជាន់ខ្ពស់, គឺប្រុស។

គូស្វាមីភរិយាដ៏ពិសិដ្ឋ, អាពាហ៍ពិពាហ៍ដ៏ទេវភាពអស់កល្បជានិច្ច, តែងតែជានិមិត្តរូបដោយត្រីពីរដែលភ្ជាប់ដោយសហសញ្ញា; ចុងក្រោយនេះគឺ អង្គ ATMAN។

គូស្វាមីភរិយាដ៏ពិសិដ្ឋ, ត្រីអស់កល្បទាំងពីរ, ធ្វើការក្នុងចំនោមទឹកនៃជ្រោះនៅពេលដែលព្រឹកព្រលឹមនៃ MAHANVANTARA មកដល់។

ត្រីដែលមិនអាចនិយាយបានទាំងពីរធ្វើការក្រោមការដឹកនាំរបស់ ATMAN នៅពេលដែលពេលវេលានៃព្រឹកព្រលឹមនៃការបង្កើតមកដល់។

ទោះជាយ៉ាងណាក៏ដោយ វាជាការល្អក្នុងការចងចាំថា អ៊ីស៊ីស និង អូស៊ីរីស មិនអាចធ្វើការនៅក្នុងស្នាដៃដ៏អស្ចារ្យបានទេ បើគ្មានបារតដ៏ល្បីល្បាញនៃទស្សនវិជ្ជាសម្ងាត់។ នៅក្នុងបារតផ្លូវភេទនេះមានគន្លឹះនៃអំណាចទាំងអស់។

រង្វង់មួយដែលមានបន្ទាត់បញ្ឈរឆ្លងកាត់ជានិមិត្តសញ្ញា ហេរ៉ាទិក, គឺជាការរួបរួមដ៏ពិសិដ្ឋបំផុតនៃភាពអស់កល្បជានិច្ចនៃស្រីជាមួយបុរសអស់កល្បជានិច្ច; បន្សំពណ៌នៃភាពផ្ទុយគ្នានៅក្នុង MONAD សំខាន់, ដែលមិនអាចនិយាយបាន និងដ៏ទេវភាព។

ពីក្នុងចំណោមលំហ ព្រះមាតាដ៏អស្ចារ្យកើតមក មនោន, អង្គ។ ពីក្នុងចំណោមមហាសមុទ្រដ៏អស្ចារ្យ ពួកអេឡូហ៊ីម ក្រោកឈរឡើងដើម្បីធ្វើការនៅព្រឹកព្រលឹមនៃ MAHANVANTARA។

ទឹក គឺជាធាតុស្រីនៃអ្វីៗទាំងអស់ដែលបានបង្កើត ពីកន្លែងដែល MATER LATINA មកពីអក្សរ M ដ៏ទេវភាពយ៉ាងខ្លាំង។

នៅក្នុងគ្រិស្តសាសនា ជីណូស្ទីក ម៉ារី គឺជាអ៊ីស៊ីស ដូចគ្នា, ព្រះមាតានៃសកលលោក, លំហ ព្រះមាតាអស់កល្បជានិច្ច, ទឹកជ្រៅនៃជ្រោះ។

ពាក្យ ម៉ារី ត្រូវបានបែងចែកជាពីរព្យាង្គ; ទីមួយគឺ MAR ដែលរំលឹកយើងអំពីមហាសមុទ្រជ្រៅនៃ ត្រី។ ទីពីរគឺ ÍA ដែលជាបំរែបំរួលនៃ IO (iiioooo), ឈ្មោះដ៏ឧត្ដុង្គឧត្ដមនៃ លំហព្រះមាតា, រង្វង់នៃភាពទទេ, ពីកន្លែងដែលអ្វីៗទាំងអស់បញ្ចេញនិងកន្លែងដែលអ្វីៗទាំងអស់ត្រឡប់មកវិញ; មួយ, តែមួយគត់នៃសកលលោកដែលបានបង្ហាញ, បន្ទាប់ពីរាត្រីនៃការបំផ្លិចបំផ្លាញដ៏អស្ចារ្យឬការបំផ្លាញ។

បំបែកទឹកខាងលើចេញពីទឹកខាងក្រោម ពន្លឺត្រូវបានបង្កើតឡើង ពោលគឺពាក្យដែលបង្កើតចលនានៃសកលលោក កូនប្រុសបានផុសឡើងក្នុងជីវិត ហើយជីវិតនេះបានយកព្រះអាទិត្យជាធាតុបញ្ជូន ដែលស្ថិតនៅចំកណ្តាលនៃប្រព័ន្ធព្រះអាទិត្យរបស់យើង ដូចជាបេះដូងនៅក្នុងរាងកាយរបស់យើង។

រំញ័រដ៏មានជីជាតិនៃព្រះអាទិត្យពិតជាភ្លើងធាតុរស់ដែលប្រមូលផ្តុំនៅចំកណ្តាលនៃភពនីមួយៗដែលបង្កើតជាបេះដូងនៃភពនីមួយៗទាំងនេះ។

ពន្លឺទាំងអស់នោះ ជីវិតទាំងអស់នោះ ត្រូវបានតំណាងដោយវិញ្ញាណទាំងប្រាំពីរនៅចំពោះបល្ល័ង្ក នៅក្នុងប្រាសាទបេះដូងនៃភពទាំងប្រាំពីរនៃប្រព័ន្ធព្រះអាទិត្យនីមួយៗ។

ការងារនៃការបំបែកទឹកចេញពីទឹកត្រូវគ្នាទៅនឹងគូស្វាមីភរិយាដ៏ពិសិដ្ឋ។ វិញ្ញាណទាំងប្រាំពីរនីមួយៗនៅចំពោះបល្ល័ង្កបានបញ្ចេញចេញពីអង្គទ្រង់ គូស្វាមីភរិយាត្រីដ៏ពិសិដ្ឋដើម្បីធ្វើការនៅព្រឹកព្រលឹមនៃការបង្កើតជាមួយនឹងអំណាចនៃ KRIYASAKTY, អំណាចនៃពាក្យដែលបាត់បង់, អំណាចនៃឆន្ទៈ និង យូហ្គា។

សេចក្ដីស្រឡាញ់នៃសេចក្ដីស្រឡាញ់, ចំណង់ចំណូលចិត្តអាថ៌កំបាំងនៃភ្លើងចុងក្រោយរវាងគូស្វាមីភរិយាអស់កល្បជានិច្ច និង ភរិយាដ៏ទេវភាពមានសារៈសំខាន់សម្រាប់ការបំបែកទឹកខាងលើចេញពីទឹកខាងក្រោម។

នៅក្នុងការងារនេះមាន ម៉ាទីណា ឆ្លងកាត់; KRIYASAKTY, ពាក្យច្នៃប្រឌិត។

គាត់នាំយកភ្លើងហើយនាងផ្លាស់ប្តូរទឹកដោយបំបែកផ្នែកខាងលើចេញពីផ្នែកខាងក្រោម។

ត្រីទាំងពីរ បន្ទាប់មកបញ្ចាំងភ្លើង និងទឹកខាងលើដែលបានផ្លាស់ប្តូរនោះទៅលើទឹកនៃភាពវឹកវរ លើរូបធាតុលោហធាតុ ឬសម្ភារៈសម្រាប់ពិភពលោក លើមេរោគដែលដេកលក់នៃអត្ថិភាព ហើយជីវិតក៏ផ្ទុះឡើង។

ការងារទាំងអស់ត្រូវបានអនុវត្តដោយមានជំនួយពីពាក្យ និងឆន្ទៈ និង យូហ្គា។

ដំបូង សកលលោកគឺស្រទន់ បន្ទាប់មកវាប្រមូលផ្តុំសម្ភារៈ ដោយឆ្លងកាត់ដំណាក់កាលបន្តបន្ទាប់គ្នានៃការគ្រីស្តាល់រីកចម្រើន។

មានសកលលោកជាច្រើនលាននៅក្នុងលំហអាកាសដ៏ធំ នៅក្នុងទ្រូងនៃលំហ ព្រះមាតា។

សកលលោកខ្លះកំពុងចេញពី PRALAYA, ផ្ទុះចេញពីទឹកជ្រៅនៃ ត្រី, ខ្លះទៀតកំពុងមានសកម្មភាពពេញលេញ ខ្លះទៀតកំពុងរលាយក្នុងទឹកអស់កល្បជានិច្ច។
