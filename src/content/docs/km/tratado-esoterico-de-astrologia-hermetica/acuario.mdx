## ពីថ្ងៃទី ២០ ខែមករា ដល់ថ្ងៃទី ១៩ ខែកុម្ភៈ

អត្ថន័យលាក់កំបាំងនៃ AQUARIUS គឺ ចំណេះដឹង។ AQUARIUS ដែលជានិមិត្តសញ្ញានៃអ្នកកាន់ទឹក គឺជាសញ្ញារាសីចក្របដិវត្តន៍ដ៏លេចធ្លោ។

មានចំណាត់ថ្នាក់ចំនួនបួននៃចំណេះដឹង ឬវិទ្យាសាស្ត្រសម្ងាត់។ យើងត្រូវដឹងថាតើចំណាត់ថ្នាក់ទាំងបួននៃចំណេះដឹងទាំងនោះជាអ្វី។

ទីមួយ៖ VAJNA-VIDYA; ចំណេះដឹងដែលទទួលបានដោយថាមពលលាក់កំបាំងជាក់លាក់ដែលបានដាស់នៅក្នុងធម្មជាតិខាងក្នុងរបស់យើងផ្ទាល់ តាមរយៈពិធីវេទមន្តមួយចំនួន។

ទីពីរ៖ MAHA-VIDYA KABALÍSTICA ។ វិទ្យាសាស្ត្រនៃ KÁBALA ជាមួយនឹងការអំពាវនាវ គណិតវិទ្យា និមិត្តសញ្ញា និងការបូជាទាំងអស់របស់វា អាចជា ANGÉLICA ឬ DIABÓLICA អ្វីគ្រប់យ៉ាងអាស្រ័យលើប្រភេទមនុស្សដែលប្រើវា។

ទីបី៖ GUPTA-VIDYA; វិទ្យាសាស្ត្រនៃ MANTRAM, វេទមន្តនៃ VERBO; វាផ្អែកលើអំណាចអាថ៌កំបាំងនៃសម្លេង នៅក្នុងវិទ្យាសាស្ត្រនៃ HARMONY ។

ទីបួន៖ ATMA-VIDYA ឬ WISDOM ពិតប្រាកដនៃ BEING នៃ ATMAN នៃ MONADA SUPERIOR ។

ទម្រង់ទាំងអស់នៃចំណេះដឹងទាំងនេះ លើកលែងតែទីបួន ដែលជាឫសគល់នៃវិទ្យាសាស្ត្រសម្ងាត់ទាំងអស់។ ចេញពីទម្រង់ចំណេះដឹងទាំងអស់នោះ លើកលែងតែទីបួន មក KÁBALA, QUIROMANCIA, ASTROLOGÍA, FISIOLOGÍA OCULTA, CARTOMANCIA CIENTÍFICA ជាដើម។

ចេញពីទម្រង់ចំណេះដឹងទាំងអស់នោះ ចេញពីសាខា OCULTISTAS ទាំងអស់នោះ វិទ្យាសាស្ត្របានរកឃើញអាថ៌កំបាំងមួយចំនួនរួចហើយ ប៉ុន្តែ SENSE SPATIAL DEVELOPED មិនមែនជាការ hypnosis ហើយក៏មិនអាចទទួលបានដោយសិល្បៈទាំងនោះដែរ។

សៀវភៅ ASTROLÓGICO HERMÉTICO ESOTÉRICO បច្ចុប្បន្ន គ្មានអ្វីដែលត្រូវធ្វើជាមួយ ASTROLOGÍA នៃពិព័រណ៍ដែលបានរៀបរាប់នៅក្នុង PERIÓDICOS ទេ។ នៅក្នុងសៀវភៅនេះ យើងបង្រៀន CIENCIA នៃ ATMA-VIDYA ។

រឿងសំខាន់គឺ ATMA-VIDYA វាមានទាំងអស់នៅក្នុងទិដ្ឋភាព ESENCIAL របស់វា ហើយថែមទាំងអាចប្រើពួកវាបានម្តងម្កាល ប៉ុន្តែវាគ្រាន់តែប្រើសារធាតុសំយោគរបស់ពួកគេដែលត្រូវបានបន្សុតពី slag ទាំងអស់។

PUERTA DE ORO នៃប្រាជ្ញាអាចប្រែទៅជាទ្វារធំទូលាយ និងផ្លូវធំទូលាយដែលនាំទៅរកការបំផ្លិចបំផ្លាញ ទ្វារនៃសិល្បៈវេទមន្តដែលអនុវត្តសម្រាប់គោលបំណងអាត្មានិយម។

យើងស្ថិតនៅក្នុង EDAD នៃ KALI-YUGA, យុគសម័យនៃ HIERRO, EDAD NEGRA ហើយសិស្សទាំងអស់នៃ OCULTISMO មានទំនោរទៅបាត់បង់តាមផ្លូវខ្មៅ។ វាគួរឱ្យភ្ញាក់ផ្អើលដែលឃើញគំនិតខុសឆ្គងយ៉ាងខ្លាំងដែល "បងប្អូនតូចៗ" មានអំពី OCULTISMO និងភាពងាយស្រួលដែលពួកគេជឿថាពួកគេអាចទៅដល់ PUERTA ហើយឆ្លងកាត់ UMBRAL នៃ MISTERIO ដោយគ្មាន SACRIFICIO ដ៏អស្ចារ្យនោះទេ។

វាមិនអាចទៅរួចទេក្នុងការសម្រេចបាននូវ ATMA-VIDYA ដោយគ្មាន TRES FACTORES នៃការបដិវត្តន៍នៃ CONCIENCIA ។

វាមិនអាចទៅរួចទេ ATMA-VIDYA ដោយមិនបានទៅដល់ NACIMIENTO SEGUNDO ។ វាមិនអាចទៅរួចទេ ATMA-VIDYA ដោយគ្មានការស្លាប់នៃ YO PLURALIZADO ។ វាមិនអាចទៅរួចទេ ATMA-VIDYA ដោយគ្មាន SACRIFICIO សម្រាប់មនុស្សជាតិ។

វាមិនមែនជាច្បាប់នៃ EVOLUCIÓN ដែលផ្តល់ឱ្យយើងនូវ ATMA- VIDYA នោះទេ។ វាមិនមែនជាច្បាប់នៃ INVOLUCIÓN ដែលផ្តល់ឱ្យយើងនូវ ATMA-VIDYA នោះទេ។ តែនៅលើមូលដ្ឋាននៃ TREMENDAS និង ESPANTOSAS REVOLUCIONES ÍNTIMAS យើងទៅដល់ ATMA-VIDYA ។

CAMINO នៃការបដិវត្តន៍នៃ CONCIENCIA គឺជា SENDA DEL FILO DE LA NAVAJA; SENDA នេះគឺ TERRIBLEMENTE DIFÍCIL; SENDA នេះគឺពោរពេញទៅដោយគ្រោះថ្នាក់នៅខាងក្នុង និងខាងក្រៅ។

ឥឡូវនេះ យើងនឹងសិក្សានៅក្នុងជំពូកនេះ នូវកត្តា TRES នីមួយៗនៃការបដិវត្តន៍នៃ CONCIENCIA តាមលំដាប់លំដោយ និងដោយឡែកពីគ្នា ដើម្បីឱ្យសិស្ស GNÓSTICOS អាចតម្រង់ទិសខ្លួនឯងបានត្រឹមត្រូវ។

ដូច្នេះ សូមឱ្យអ្នកអានរបស់យើងយកចិត្តទុកដាក់យ៉ាងខ្លាំងក្នុងការសិក្សាកត្តា TRES នីមួយៗ នៃការបដិវត្តន៍នៃ CONCIENCIA ព្រោះភាពជោគជ័យនៅក្នុង TRABAJO នេះអាស្រ័យលើការយល់ដឹងពេញលេញនៃកត្តា TRES នីមួយៗទាំងនេះ។

### កើត

NACIMIENTO SEGUNDO គឺជាបញ្ហាផ្លូវភេទទាំងស្រុង។ TORO APIS ដ៏ពិសិដ្ឋក្នុងចំណោមជនជាតិ EGIPCIOS ជំនាន់ដើម ត្រូវតែនៅក្មេង មានសុខភាពល្អ និងរឹងមាំ ដើម្បីតំណាងឱ្យ PIEDRA FILOSOFAL ។ (ភេទ។)

ជនជាតិ Griegos ដែលត្រូវបានណែនាំដោយ Hierofantes Egipcios បានតំណាងឱ្យ PIEDRA FILOSOFAL ជាមួយនឹង TOROS មួយ ឬច្រើន ដូចដែលយើងឃើញផងដែរនៅក្នុងរឿងនិទានរបស់ MINOTAURO CRETENSE ។

Alkímica ដូចគ្នាក៏មានន័យដូចគ្នាចំពោះ TOROS ដែល HÉRCULES បានលួចពី Gerión និមិត្តសញ្ញាដូចគ្នានេះយើងបានរកឃើញនៅក្នុងរឿងព្រេងនៃ SAGRADOS BUEYES នៃព្រះអាទិត្យដែលចិញ្ចឹមដោយស្ងប់ស្ងាត់នៅលើ ISLA នៃ SICILIA ហើយត្រូវបានលួចដោយ Mercurio ។

TOROS ដ៏ពិសិដ្ឋទាំងអស់មិនមែនសុទ្ធតែខ្មៅ ឬសនោះទេ; ខ្លះមាន ROJOS ដូចរបស់ Gerión និងដូចជាអ្នកដែលត្រូវបានបូជាដោយបូជាចារ្យ Israelita ព្រោះ PIEDRA FILOSOFAL នៅពេល Alkímico ជាក់លាក់មួយ គឺ ROJA ហើយនេះត្រូវបានគេដឹងដោយ Alkimista ទាំងអស់។

គោ APIS ដ៏ល្បីល្បាញ ដែលត្រូវបានគោរពបូជាយ៉ាងខ្លាំងនៅក្នុង MISTERIOS EGIPCIOS គឺជាអ្នកបង្កើត និងជាព្រះរាជអាជ្ញានៃ ALMAS ។ គោ APIS SIMBÓLICO ត្រូវបានឧទ្ទិសដល់ ISIS ព្រោះតាមពិតវាទាក់ទងនឹង VACA SAGRADA, MADRE DIVINA, ISIS ដែលគ្មាននរណាម្នាក់លើកស្បៃមុខ។

ដើម្បីឱ្យគោមួយមានកិត្តិយសខ្ពស់ក្នុងការឡើងទៅកាន់ឋានៈបែបនេះ វាចាំបាច់ត្រូវតែខ្មៅ ហើយមានស្នាមពណ៌សនៅលើថ្ងាស ឬនៅលើបន្ទះមួយក្នុងទម្រង់ជាអឌ្ឍចន្ទ Lunar ។

វាក៏ពិត និងពិតទាំងស្រុងដែលគោដ៏ពិសិដ្ឋនោះត្រូវតែត្រូវបានបង្កើតឡើងក្រោមការចាប់អារម្មណ៍នៃកាំរស្មី ហើយមានសញ្ញានៃសត្វកណ្ដៀរពិសិដ្ឋនៅក្រោមអណ្ដាត។

APIS គឺជានិមិត្តសញ្ញានៃ LUNA ទាំងដោយសារតែស្នែងរបស់វាមានរាងជាអឌ្ឍចន្ទ Lunar និងដោយសារតែក្នុងអំឡុងពេល លើកលែងតែ Plenilunio តារាសាស្ត្រនេះតែងតែមានផ្នែកងងឹតដែលបង្ហាញដោយពណ៌ខ្មៅនៃស្បែក និងផ្នែកភ្លឺដែលតំណាងដោយស្នាមពណ៌ស។

APIS គឺជា MATERIA FILOSOFAL ENS SEMINIS (SEMEN) សារធាតុ semisolid semilíquida VITRIOLO នៃ ALKIMISTAS ។
