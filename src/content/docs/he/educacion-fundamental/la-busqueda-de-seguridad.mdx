כאשר האפרוחים מפחדים הם מתחבאים מתחת לכנפיה האוהבות של התרנגולת בחיפוש אחר ביטחון.

הילד המפוחד רץ לחפש את אמו כי לידה הוא מאמין שהוא בטוח.

לפיכך, הוכח כי פחד והחיפוש אחר ביטחון קשורים תמיד זה בזה באופן אינטימי.

האיש שחושש שודדים מחפש ביטחון באקדח שלו.

המדינה שחוששת שארץ אחרת תתקוף אותה, תקנה תותחים, מטוסים, ספינות מלחמה ותחמש צבאות ותצא למלחמה.

אנשים רבים שאינם יודעים לעבוד, מבוהלים מפני הסבל מחפשים ביטחון בפשע, והופכים לגנבים, תוקפים וכו'...

נשים רבות חסרות אינטליגנציה המפוחדות מפני האפשרות של סבל הופכות לזונות.

האיש הקנאי חושש לאבד את אשתו ומחפש ביטחון באקדח, הורג ואז ברור שהוא יגיע לכלא.

האישה הקנאית הורגת את יריבתה או את בעלה וכך הופכת לרוצחת.

היא חוששת לאבד את בעלה ורוצה להבטיח אותו לעצמה, הורגת את האחרת או מחליטה להרוג אותו.

בעל הבית שחושש שאנשים לא ישלמו לו את שכר הדירה של הבית, דורש חוזים, ערבים, פיקדונות וכו', ובכך רוצה להבטיח את עצמו, ואם אלמנה ענייה ומלאה בילדים לא יכולה למלא דרישות אדירות כאלה, ואם כל בעלי הבתים בעיר עושים את אותו הדבר, בסופו של דבר האומללה תצטרך ללכת לישון עם ילדיה ברחוב או בפארקים של העיר.

כל המלחמות מקורן בפחד.

הגסטאפו, העינויים, מחנות הריכוז, סיביר, בתי הסוהר הנוראים, הגלויות, עבודת הכפייה, ההוצאות להורג וכו' מקורם בפחד.

אומות תוקפות אומות אחרות מפחד; הן מחפשות ביטחון באלימות, הן מאמינות שעל ידי הריגה, פלישה וכו' הן יוכלו להיות בטוחות, חזקות ועוצמתיות.

במשרדי המשטרה החשאית, הריגול הנגדי וכו', הן במזרח והן במערב, מענים מרגלים, חוששים מהם, רוצים לגרום להם להודות במטרה למצוא ביטחון למדינה.

כל הפשעים, כל המלחמות, כל הפשעים, מקורם בפחד ובחיפוש אחר ביטחון.

בעבר הייתה כנות בין האנשים, כיום הפחד והחיפוש אחר ביטחון חיסלו את הניחוח הנפלא של הכנות.

החבר לא בוטח בחבר, הוא חושש שהוא יגנוב אותו, ירמה אותו, ינצל אותו ואפילו יש אמירות טיפשיות וסוטות כמו: "לעולם אל תפנה את גבך לחבר הכי טוב שלך". ההיטלריסטים אמרו שהאמירה הזו היא זהב.

החבר כבר מפחד מהחבר ואפילו משתמש באמרות כדי להגן על עצמו. כבר אין כנות בין חברים. הפחד והחיפוש אחר ביטחון חיסלו את הניחוח הטעים של הכנות.

קסטרו רוס בקובה הוציא להורג אלפי אזרחים מחשש שיסיימו אותו; קסטרו מחפש ביטחון על ידי הוצאה להורג. הוא מאמין שכך הוא יכול למצוא ביטחון.

סטלין, סטלין הסוטה והצמא דם, הפיץ את רוסיה עם הטיהורים העקובים מדם שלו. זו הייתה הדרך לחפש את ביטחונו.

היטלר ארגן את הגסטאפו, הגסטאפו הנורא, לביטחון המדינה. אין ספק שהוא חשש שידיחו אותו ולכן ייסד את הגסטאפו העקוב מדם.

כל המרירות בעולם הזה מקורה בפחד ובחיפוש אחר ביטחון.

מורים ומורות צריכים ללמד את התלמידים והתלמידות את מעלת האומץ.

חבל שממלאים ילדים וילדות בפחד מביתם.

ילדים וילדות מאוימים, מוטרדים, מפוחדים, מוכים וכו'.

נהוג שאבות משפחה ומורים להפחיד את הילד והצעיר כדי שילמד.

בדרך כלל אומרים לילדים וצעירים שאם לא ילמדו יצטרכו לקבץ נדבות, לשוטט רעבים ברחובות, לעסוק בעבודות צנועות מאוד כמו צחצוח נעליים, נשיאת חבילות, מכירת עיתונים, עבודה בחריש וכו' וכו' וכו'. (כאילו עבודה היא פשע)

בעומק, מאחורי כל המילים האלה של הורים ומורים, יש פחד על הבן וחיפוש אחר ביטחון עבור הבן.

הדבר החמור בכל מה שאנחנו אומרים הוא שהילד והצעיר מסובכים, מתמלאים בפחד ומאוחר יותר בחיים המעשיים הם אנשים מלאי פחד.

הורים ומורים שיש להם את הטעם הרע להפחיד ילדים וילדות, צעירים וצעירות, באופן לא מודע מובילים אותם בדרך הפשע, שכן כפי שאמרנו, לכל פשע יש מקורו בפחד ובחיפוש אחר ביטחון.

כיום הפחד והחיפוש אחר ביטחון הפכו את כדור הארץ לגיהנום נורא. כולם מפחדים. כולם רוצים ביטחון.

בעבר אפשר היה לנסוע בחופשיות, כיום הגבולות מלאים בשומרים חמושים, נדרשים דרכונים ותעודות מכל הסוגים כדי לקבל זכות לעבור ממדינה למדינה.

כל זה הוא תוצאה של פחד וחיפוש אחר ביטחון. מפחדים ממי שנוסע, מפחדים ממי שמגיע ומחפשים ביטחון בדרכונים ובניירות מכל הסוגים.

מורים ומורות בבתי ספר, מכללות, אוניברסיטאות צריכים להבין את הזוועה של כל זה ולשתף פעולה לטובת העולם, בידיעה כיצד לחנך את הדורות החדשים, ללמד אותם את דרך האומץ האמיתי.

דחוף ללמד את הדורות החדשים לא לפחד ולא לחפש ביטחון בשום דבר ובאף אחד.

חיוני שכל אדם ילמד לבטוח יותר בעצמו.

הפחד והחיפוש אחר ביטחון הם חולשות נוראיות שהפכו את החיים לגיהנום נורא.

בכל מקום שופעים פחדנים, פחדנים, חלשים שתמיד מחפשים ביטחון.

מפחדים מהחיים, מפחדים מהמוות, מפחדים ממה יגידו, "מה אומרים שאומרים", מלאבד את המעמד החברתי, את המעמד הפוליטי, את היוקרה, את הכסף, את הבית היפה, את האישה היפה, את הבעל הטוב, את התעסוקה, את העסק, את המונופול, את הרהיטים, את המכונית וכו' וכו' וכו', מפחדים מהכל, שופעים בכל מקום פחדנים, פחדנים, חלשים, אבל אף אחד לא מאמין שהוא פחדן, כולם מתיימרים להיות חזקים, אמיצים וכו'.

בכל המעמדות החברתיים קיימים אלפי ומיליוני אינטרסים שמפחדים לאבד ובגלל זה כולם מחפשים ביטחונות שבאמצעות הפיכתם ליותר ויותר מורכבים, הופכים את החיים ליותר ויותר מסובכים, יותר ויותר קשים, יותר ויותר מרים, אכזריים וחסרי רחמים.

כל הרכילויות, כל ההשמצות, המזימות וכו', מקורן בפחד ובחיפוש אחר ביטחון.

כדי לא לאבד את ההון, המעמד, הכוח, היוקרה, מפיצים השמצות, רכילויות, רוצחים, משלמים כדי שירצחו בסתר וכו'.

לעשירים בעולם יש אפילו את המותרות של החזקת רוצחים בשכר ומשלמים להם היטב, במטרה דוחה לחסל את כל מי שמאיים להאפיל עליהם.

הם אוהבים את הכוח בשביל הכוח עצמו ומבטיחים אותו על בסיס כסף והרבה דם.

העיתונים מפרסמים כל הזמן חדשות על מקרים רבים של התאבדות.

רבים מאמינים שמי שמתאבד הוא אמיץ, אבל באמת מי שמתאבד הוא פחדן שמפחד מהחיים ומחפש ביטחון בזרועות המוות המצומקות.

כמה גיבורי מלחמה נודעו כאנשים חלשים ופחדנים, אבל כשעמדו פנים אל פנים עם המוות, האימה שלהם הייתה כה נוראה, שהם הפכו לחיות טרף נוראות שחיפשו ביטחון לחייהם, ועשו מאמץ עליון נגד המוות. אז הם הוכרזו כגיבורים.

לעתים קרובות טועים בין פחד לאומץ. מי שמתאבד נראה מאוד אמיץ, מי שנושא אקדח נראה מאוד אמיץ, אבל במציאות המתאבדים והאקדוחנים הם מאוד פחדנים.

מי שלא מפחד מהחיים לא מתאבד. מי שלא מפחד מאף אחד לא נושא אקדח על המותן.

דחוף שמורים ומורות ילמדו את האזרח בצורה ברורה ומדויקת מהו ערך אמיתי ומהו פחד.

הפחד והחיפוש אחר ביטחון הפכו את העולם לגיהנום נורא.
